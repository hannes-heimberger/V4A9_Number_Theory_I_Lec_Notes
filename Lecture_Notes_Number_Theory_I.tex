\begin{definition}{$\Zprim{n}$}{1.0}
  We denote
  \[ \index{$\Zprim{n}$} {\color{red} \Zprim{n}} = \{ (x_1, \dots, x_n) \in \Z^n : \gcd(x_1, \dots, x_n) = 1\}.\]
\end{definition}
\begin{theorem}{Dirichlet, 1842}{1.1}
  Let $\alpha \in \R$. Then, for every integer $Q \geq 2$, there exists $(x,y) \in \Zprim{2}$ such that $0 < y \leq Q$ and $|x-\alpha y| \leq \nicefrac{1}{Q}$.
\end{theorem}
\begin{pf}
  Consider the $Q+1$ numbers $0, \alpha-\lf\alpha\rf, 2\alpha-\lf 2\alpha\rf, \dots, Q\alpha-\lf Q\alpha\rf$. Each of these is in the interval $[0,1)$. Partition the interval $[0,1)$ into $Q$ subintervals of length $\nicefrac{1}{Q}$. Now, we can apply the pigeonhole principle to deduce that
  \[\exists_{0 \leq i \leq j \leq Q}: |(j\alpha-\lf j\alpha\rf)-(i\alpha-\lf i\alpha\rf)| \leq \frac{1}{Q}.\]
  This implies
  \[ |(j-i)\alpha - (\lf j\alpha\rf -\lf i\alpha\rf)| \leq \frac{1}{Q}.\]
  Let $\widetilde{y} = j-i$ and $\widetilde{x} = \lf j\alpha\rf - \lf i\alpha \rf$. Then $|\widetilde{y}\alpha-\widetilde{x}| \leq \nicefrac{1}{Q}$. Let $d = (\widetilde{y}, \widetilde{x})$, $y = \nicefrac{\widetilde{y}}{d}$ and $x = \nicefrac{\widetilde{x}}{d}$ and divide through by d to get that
  \[ |y\alpha-x| \leq \frac{1}{dQ} \leq \frac{1}{Q}. \qedhere\]
\end{pf}
A very simple consequence of this:
\begin{corollary}{Irrational approximation}{1.2}
  Let $\alpha \in \R \setminus \Q$. Then there exist infinitely many $(x,y) \in \Zprim{2}$ such that $y > 0$ and
  \[ \left|\alpha-\frac{x}{y}\right| \leq \frac{1}{y^2}.\]
\end{corollary}
\begin{pf}
  For each $Q \geq 2$, let $(x_Q, y_Q) \in \Zprim{2}$ be such that $0 < y_Q \leq Q$ and $|x_Q-y_Q\alpha| \leq \nicefrac{1}{Q}$ (Theorem \ref{thm:1.1} ensures the existence of such a pair). This implies
  \[ \left| \alpha-\frac{x_Q}{y_Q}\right| \leq \frac{1}{Qy_Q} \leq \frac{1}{y_Q^2}.\]
  Suppose the set
  \[ S = \{ (x_Q,y_Q): Q \geq 2\}\]
  is finite. Then there exists $(x_0,y_0) \in \Zprim{2}$ such that $(x_Q,y_Q) = (x_0,y_0) \jubelforall_{Q \geq Q_0}$ and some $Q_0$. This means
  \[ \left| \alpha-\frac{x_Q}{y_Q}\right| = \left| \alpha - \frac{x_0}{y_0}\right| \leq \frac{1}{Q} \jubelforall_{Q \geq Q_0}.\]
  Taking the limit as $Q \to \infty$ yields
  \[ \alpha = \frac{x_0}{y_0},\]
  which is a contradiction due to the fact that $\alpha$ was supposed to be irrational. \Lightning
\end{pf}
\begin{question}{Is this error term optimal?}{is_opt_irr_apt}
  Can we do better than $\nicefrac{1}{y^2}$ for some $\alpha \in \R$? More explicitly, do there exist infinitely many $(x,y) \in \Zprim{2}$ such that $y > 0$ and
  \[ \exists_{\delta > 0}:\left| \alpha - \frac{x}{y}\right| < \frac{1}{y^{2+\delta}}.\]
\end{question}
\begin{answertoquestion} Yes, but it is rare. However, this is not the case if $\alpha$ is algebraic over $\Q$.
\end{answertoquestion}

 \section{Lattices and discrete subgroups}
 Let $n \geq 2$. Given $x_1, \dots, x_n \in \R^n$, we say that they are linearly independent over $\R$, if $c_1x_1 + \dots + c_rx_r \neq 0 \forall_{(c_1, \dots, c_r)^t \in \R^r \setminus \{ 0\}}$. We consider the usual topology on $\R^n$ generated by the Euclidean metric. For subsets of $\R^n$, we consider the subspace topology. We say a subset $\Lambda \subseteq \R^n$ is \emp{discrete}\index{Discrete Subset of $\R^n$} if $\#\Lambda \cap B < \aleph_0$ for every bounded subset $B \subseteq \R^n$. Furthermore, $\Lambda$ is a \define[Discrete Subgroup of $\R^n$]{discrete subgroup} if it is discrete and $c_1x_1 + c_2x_2 \in \Lambda \forall_{c_1,c_2 \in \Z, x_1,x_2 \in \Lambda}$. In this case, we define \index{Rank of a Discrete Subgroup of $\R^n$}${\color{red} \rank(\Lambda)} = r$ as the maximal number $r$ such that $\Lambda$ contains $r$ linearly independent vectors over $\R$. A \define[Basis of a Discrete Subgroup of $\R^n$]{basis} of $\Lambda$ is a set $\{ v_1, \dots, v_r\}$ of linearly independent vectors over $\R$ such that $\Lambda = \{ c_1v_1 + \dots + c_rv_r: c_1, \dots, c_r \in \Z\}$.
 \begin{lemma}{Lattice Basis Construction}{2.1}
   Let $\Lambda$ be a discrete subgroup of $\R^n$ and $\rank(\Lambda) = r \geq 1$. Also assume that $\{ w_1, \dots, w_r\} \subseteq \Lambda$ is linearly independent over $\R$. Then there exists a basis $\{ v_1, \dots, v_r\}$ of $\Lambda$ such that for all $1 \leq k \leq r$, we have
   \[ v_k = \alpha_{k,1}w_1 + \dots + \alpha_{k,k}w_k \text{ for some } \alpha_{k,1}, \dots, \alpha_{k, k} \in [0,1] \text{ with } \alpha_{k,k} \neq 0.\]
 \end{lemma}
 \begin{pf}
   Let $1 \leq k \leq r$. We define
   \[ S_k = \{ \alpha_1w_1 + \dots + \alpha_kw_k: \alpha_1, \dots, \alpha_k \in [0,1], \alpha_k \neq 0\}.\]
   Since $\Lambda$ is discrete and $S_k$ is bounded,
   \[ \underbrace{0 <}_{w_k \in \Lambda \cap S_k} \#(\Lambda \cap S_k) \ll 1.\]
   We choose $v_k$ to be a vector from $\Lambda \cap S_k$ with minimal $\alpha_k$. Let us write
   \[ v_k = \alpha_{k,1}w_1 + \dots + \alpha_{k,k}w_k.\]
   Then for any
    \[ \alpha_1w_1 + \dots + \alpha_kw_k \in \Lambda \cap S_k,\]
    we have $0 < \alpha_{k,k} \leq \alpha_k \leq 1$. It is clear that $\{ v_1, \dots, v_r\}$ is linearly independent over $\R$. For each $0 \leq k \leq r$, let
    \[ M_k = \Lambda \cap \left\{ \sum_{j=1}^k \alpha_jw_j: \alpha_1, \dots, \alpha_k \in \R\right\} \text{ with } M_0 = \{ 0\}.\]
    Note that $M_r = \Lambda$. We now use induction to show that $M_k = \{ c_1v_1 + \dots + c_kv_k: c_1, \dots, c_k \in \Z\} \forall_{0 \leq k \leq r}$. When $k = 0$, the statement is trivially true. Suppose the statement holds for $k-1$. \\
    Let
    \[ x = \sum_{j=1}^k \beta_jw_j \in M_k \subseteq \Lambda \text{ and set } z_k = \lf \frac{\beta_k}{\alpha_{k,k}}\rf \in \Z.\]
    Then, $0 \leq \beta_k - z_k\alpha_{k,k} < \alpha_{k,k}$. Consider $x - z_k\underbrace{v_k}_{\in \Lambda} \in \Lambda$. Also, \begin{align*}
    & x-z_kv_k = \sum_{j=1}^k (\beta_j-z_k\alpha_{k,j})w_j \\
    = \quad & \underbrace{\sum_{j=1}^{k-1} \lf \beta_j-z_k\alpha_{k,j}\rf w_j}_{\in \Lambda} + u,\end{align*}
    where
    \[ u = \sum_{j=1}^{k-1} \{\beta_j-z_k\alpha_{k,j}\}w_j+(\beta_k-z_k\alpha_{k,k})w_k.\]
    It is clear that $u \in \Lambda$. Also, if $\beta_k-z_k\alpha_{k,k} > 0$, then $u \in S_k$ (because $\{ \beta_j-z_k\alpha_{k,j}\} \in [0,1]$ and $0 < \beta_k-z_k\alpha_{k,k} < \alpha_{k,k} \leq 1$), which implies $u \in \Lambda \cap S_k$. But this contradicts the minimality of $\alpha_{k,k}$ \Lightning. Hence, $\beta_k = z_k\alpha_{k,k}$. Then,
    \[ x-z_kv_k = \sum_{j=1}^{k-1} (\beta_j-z_k\alpha_{k,j})w_j \in M_{k-1}.\]
    This concludes the induction.
 \end{pf}
 \begin{definition}{(Full) Lattice}{full_lat}
   A \define[(Full) Lattice]{(full) lattice} in $\R^n$ is a discrete sugroup of $\R^n$ which has rank $n$.
 \end{definition}
 \noindent \textbf{Remark:} For our intents and purposes, a \emp{lattice} is always \emp{full}.
 \titremark{Bases of Lattices} If $\Lambda$ is a lattice in $\R^n$, then by the previous lemma, there exists a basis $\{ v_1, \dots, v_n\}$ such that $\Lambda = \{ c_1v_1 + \dots + c_nv_n: c_1, \dots, c_n \in \Z\}$.
 \begin{definition}{Determinant of a Lattice}{det_lat}
   We define the \define[Determinant of a Lattice]{determinant} of $\Lambda$ to be
   \[ \det(\Lambda) = \left| \det(v_1 \cdots v_n)\right|.\]
 \end{definition}
 \noindent Recall that $\GL_n(\Z)$ is the multiplicative group of $n \times n$-matrices over $\Z$ with determinant $\pm 1$.
 \begin{lemma}{Independence of the Determinant of a Lattice from the Basis Choice}{2.2}
   Let $\Lambda \subseteq \R^n$ be a lattice. Suppose that $\{ v_1, \dots, v_n\}$ and $\{ w_1, \dots, w_n\}$ are two bases of $\Lambda$. Then there exists $u = (u_{i,j})_{i,j \in \{ 1, \dots, n\}} \in \GL_n(\Z)$ such that
   \[ w_i = \sum_{j=1}^n u_{i,j}v_j \forall_{1 \leq i \leq n}.\]
   In particular,
   \[ |\det(w_1 \cdots w_n)| = |\det(v_1 \cdots v_n)|.\]
 \end{lemma}
 \begin{pf}
   Let us define $U = (u_{i,j})_{i,j \in \{ 1, \dots, n\}}$ by
   \[ w_i = \sum_{j=1}^n u_{i,j}v_j \forall_{1 \leq i \leq n}.\]
   Since $w_i \in \Lambda$ and $\{ v_1, \dots, v_n\}$ is a basis of $\Lambda$, we get $u_{i,j} \in \Z \forall_{1 \leq i,j \leq n}$, which ensures $\det(U)\in \Z$. By the same argument, we define $\widetilde{U} = (\widetilde{u}_{i,j})_{i,j \in \{ 1, \dots, n\}}$ analogously. Note that $U^{-1} = \widetilde{U}$. Hence, $\det(\widetilde{U}) = \det(U) = \pm1$.
 \end{pf}
 \noindent Let $M,L$ be two lattices in $\R^n$ with $M \subseteq L$. Further let $\{ v_1, \dots, v_n\}$ be a basis of $L$ and $\{ w_1, \dots, w_n\}$ be a basis of $M$. Define $U_0$ as the $n\times n$ matrix that fulfils
 \[ (w_1 \cdots w_n) = (v_1 \cdots v_n)U_0.\]
 Then, we have $U_0 \in M_n(\Z)$. We define the \define[Index of a Sublattice in a Lattice]{index} of $M$ in $L$ by
 \[ [L:M] = |\det U_0| = \left|\frac{\det M}{\det L}\right|.\]
 \emp{Note:} This does not depend on the choice of bases of $M$ and $L$. \\
\begin{example} Let
 \[ \Lambda_0 = \Span_\Z\left\{ \binom{1}{0}, \binom{0}{1}\right\} \text{ and } \Lambda = \Span_\Z\left\{ \binom{2}{0}, \binom{0}{2}\right\}.\]
 {\color{blue} It is clear that
 \[ U_0 = \begin{pmatrix}
   2 & 0 \\
   0 & 2
 \end{pmatrix}\]
 here. We thus have
 \[ [\Lambda: \Lambda_0] = |\det U_0| = 4,\]
 which aligns nicely with the fact that every fourth point $\lambda \in \Lambda_0$ is also in $\Lambda$.}
\end{example}

\section{Convex Bodies}
\begin{definition}{Convex Subset of $\R^n$}{conv_subset_rn}
  A set $C \subseteq \R^n$ is \define[Convex Subset of $\R^n$]{convex} if given any $x,y \in C$, we have
  \[ \{ tx+(1-t)y: t \in [0,1]\} \subseteq C.\]
\end{definition}
\begin{definition}{Central Symmetric Convex Body in $\R^n$}{cent_symm_conv_bod_rn}
  A \define[Central Symmetric Convex Body in $\R^n$]{central symmetric convex body} in $\R^n$ is a closed bounded convex $C \subseteq \R^n$ fulfilling $0 \in C^\circ$ and $x \in C \iff -x\in C$.
\end{definition}
\begin{example}
  Consider any norm on $\R^n$. Then the set
  \[ B_1(0) = \{ x \in \R^n: \|x\| \leq 1\}\]
  is a central symmetric convex body.
\end{example}
\noindent Let $C \subseteq \R^n$ be a central symmetric convex body. Define
\[ \index{$C$-norm $\|x\|_C$}{\color{red} \|x\|_C} = \min\{ \lambda \in \R_{\geq 0}: x \in \lambda C\},\]
where $\lambda C = \{ \lambda x: x \in C\}$.
\begin{lemma}{Norm properties of $\|\cdot\|_C$}{norm_props_c_n}
  The map $\|\cdot\|_C$ is
  \begin{enumerate}[(i)]
    \item well-defined,
    \item a norm on $\R^n$ and
    \item $\lambda C = \{ x \in \R^n: \|x\|_C \leq \lambda\}$.
  \end{enumerate}
\end{lemma}
\begin{pf}
  We'll prove (i). Proving the other two statements is left to the reader as an exercise. Clearly, $\|0\|_C = 0$. Let $x \in \R^n \setminus \{ 0\}$. Consider the set
  \[ S = \{ \lambda \in \R_{\geq 0}: x \in \lambda C\}.\]
  Since $0$ is in the interior of $C$, there exists $\delta > 0$ such that
  \[ \{ z \in \R^n: \|z\|_2 \leq \delta\} \subseteq C.\]
  Hence, we have
  \[ x \in \frac{\|x\|_2}{\delta}C \text{, which implies } S \neq \varnothing.\]
  Let $0 \leq \mu = \inf_{\lambda \in S} \lambda$. By the definition of the infimum, we have
  \[ x \in \left(\mu + \frac{1}{m}\right)C \forall_{m \in \N}.\]
  Thus, we have
  \[ \left(\mu + \frac{1}{m}\right)^{-1}x \in C.\]
  Since $C$ is bounded, it must be that $\mu > 0$. Furthermore, since $C$ is closed and
  \[ \left(\mu+\frac{1}{m}\right)^{-1}x \in C \forall_{m \in \N},\]
  we have
  \[ \lim_{m \to \infty}\left(\mu+\frac{1}{m}\right)^{-1}x = \frac{1}{\mu}x \in C,\]
  which implies $x \in \mu C$, i.e. $\mu \in S$.
  \end{pf}
  \section{Minkowski's First Theorem}
  Given a measurable $S \subseteq \R^n$, we define the $n$-dimensional volume $\vol(S) \in \overline{\R_{\geq 0}}$ as $\lambda^n(S)$, the $n$-dimensional Lebesgue measure of $S$. \\
  \textbf{Some key properties:}
  \begin{itemize}
    \item All open and closed sets are measurable.
    \item Bounded measurable sets have finite volume.
    \item For measurable $S$,
    \[ \vol(S) = \int_S 1 \dl{x_1}\cdots\dl{x_n},\]
    if this integral (the $n$-fold Riemann integral) is defined.
    \item The set of all measurable sets in $\R^n$ forms a $\sigma$-algebra over $\R$.
     \item If $\phi: \R^n \to \R^n$ is a linear transformation and $S \subseteq \R^n$ is measurable, then $\phi(S)$ is measurable with
     \[ \vol(\phi(S)) = |\det\phi|\vol(S).\]
     \item The function $\vol$ is $\sigma$-additive.
  \end{itemize}
  Let $\Lambda \subseteq \R^n$ be a lattice with basis $\{ v_1, \dots, v_n\}$. We call
  \[ F = \{ \alpha_1v_1+ \dots + \alpha_nv_n: 0 \leq \alpha_1, \dots, \alpha_n < 1\}\]
   a \define[Fundamental Parallel Piped of $\Lambda$]{fundamental parallel} piped of $\Lambda$ (or a \define[Fundamental Domain]{fundamental domain}). We have
  \[ \R^n = \bigcup_{u \in \Lambda} u+F,\]
  where the $u+F$ are pairwise disjoint. Also,
  \[ \vol(F) = \int_{[0,1)^n} |\det[v_1 \cdots v_n]|\dl{\alpha_1}\cdots\dl{\alpha_n} = \det\Lambda.\]
  \begin{theorem}{Minkowski's First Convex Body Theorem, 1896}{mink_first_conv_bod_thm}
    Let $C$ be a central symmetric convex body in $\R^n$. Also assume that $\Lambda \subseteq \R^n$ is a lattice. Suppose we have
    \[ \vol(C) \geq 2^n\det(\Lambda).\]
    Then $C \cap (\Lambda \setminus \{ 0\}) \neq \varnothing$.
  \end{theorem}
  \begin{pf}
    First assume that $\vol(C) > 2^n\det(\Lambda)$. Then the set $\nicefrac{1}{2}C$ satisfies
    \[ \vol\left(\frac{1}{2}C\right) = \frac{1}{2^n}\vol(C) > \det\Lambda.\]
    Let $F$ be a fundamental parallel piped of $\Lambda$ and
    \[ S_a = \frac{1}{2}C \cap (a+F) \forall_{a \in \Lambda}.\]
    Then,
    \[ \sum_{a \in \Lambda}\vol(S_a) = \vol\left(\frac{1}{2}C\right) > \det\Lambda.\]
    Let us define
    \begin{align*} T_a = \quad & -a+S_a \\
    = \quad & \left(-a+\frac{1}{2}C\right) \cap F \subseteq F.
    \end{align*}
    Now,
    \[ \sum_{a \in \Lambda}\vol(T_a) = \sum_{a \in \Lambda}\vol(S_a) > \det\Lambda = \vol(F).\]
    We proved the existence of a collection of subsets of $F$ whose sum of volumes is greater than $\vol(F)$. Therefore, there exists $a \neq b \in \Lambda$ such that $T_a \cap T_b \neq \varnothing$. Let $u \in T_a \cap T_b$. There exist $x,y \in \nicefrac{1}{2}C$ such that $x-a=u=y-b$. \\
    This implies that $x-y = a-b \in \Lambda \setminus \{ 0\}$. We also know that $2x,2y \in C$. Hence, we have $-2y \in C$, which implies
    \[ \frac{1}{2}(2x)+\frac{1}{2}(-2y) = x-y \in C.\]
    Next, let us suppose $\vol(C) = 2^n\det(\Lambda)$. Let $m \in \N$. Since
    \[ \vol\left(\left(1+\frac{1}{m}\right)C\right) > 2^n\det(\Lambda),\]
    it follows from the previous case that there exists
    \[ 0 \neq x_m \in (1+\nicefrac{1}{m})C \cap \Lambda \subseteq 2C\cap\Lambda \forall_{m \in \N}.\]
    We know that $2C$ is bounded, so $2C \cap \Lambda$ is finite. Therefore, there exists an infinite sequence $m_1 < \cdots$ such that $0 \neq x_0 = x_{m_1} = x_{m_2} = \dots$. Since $x_0 = x_{m_i} \in ((1+\nicefrac{1}{m_i})C) \cap \Lambda$, we have
    \[ \left(1+\frac{1}{m_i}\right)^{-1}x_0 \in C \forall_{i \in \N}.\]
    Hence, $0 \neq x_0 \in \Lambda \cap C$.
  \end{pf}
  \begin{lemma}{Cardinality Formula for Lattice-Body Intersections}{card_form_lat_bod_int}
    Let $C$ be a central symmetric convex body and $\Lambda$ a full lattice in $\R^n$. Then
    \[ \#(xC \cap \Lambda) = \frac{\vol(C)}{\det(\Lambda)}x^n + \O(x^{n-1}) \forall_{x \geq 1}.\]
  \end{lemma}
  \begin{pf}
    Let $N(x) = \#(xC\cap \Lambda)$. Consider
    \[ S = \bigcup_{u \in xC \cap \Lambda} u+F,\]
    where $F$ is the fundamental domain of $\Lambda$.
    Since this is a disjoint union, we have
    \[ \vol(S) = N(x)\vol(F).\]
    As $F$ is bounded,
    \[ \exists_{\sigma > 0}: \|y\|_C \leq \sigma \forall_{y \in F}.\]

  \section{Minkowski's Second Theorem}
  Let $\Lambda$ be a lattice and $C$ a central symmetric convex body in $\R^n$. The \define[Successive Minima of a Lattice with respect to a Central Symmetric Convex Body]{successive minima} $0 < \lambda_1 \leq \lambda_2 \leq \dots \leq \lambda_n$ of $\Lambda$ with respect to $C$ are defined as
  \[ \lambda_i = \min\{ \lambda \in \R_{\geq 0}: \dim\Span_\R(\lambda C \cap \Lambda) \geq i\} \forall_{1 \leq i \leq n}.\]
  \begin{lemma}{Well-definedness and Construction of the Successive Minima}{2.8}
    The successive $0 < \lambda_1 \leq \dots \leq \lambda_n < \infty$ are well-defined. Also, there are $v_1, \dots, v_n \in \Lambda$, which are linearly independent over $\R$, such that $v_i \in \lambda_iC \forall_{1 \leq i \leq n}$.
  \end{lemma}
  \noindent \emp{Note:} The set $\{ v_1, \dots, v_n\}$ might not be a basis of $\Lambda$.
  \begin{pf}
    Recall that
    \[ \|x\|_C = \min\{ \lambda \in \R_{\geq 0}: x \in \lambda C\} \text{ and } \lambda C = \{ x \in \R^n: \|x\|_C \leq \lambda\}.\]
    Let $\Lambda \setminus \{ 0\} = \{ x_1, \dots\}$, where $0 < \| x_1 \|_C \leq \dots$. Also note that given any $m \in \N$, there are finitely many $i$ such that $m-1 \leq \|x_i\|_C \leq m$ (because $mC$ is bounded and $\lambda$ is discrete). Let $\lambda_1 = \|x_1\|_C$. For $i\in \{ 2, \dots, n\}$, define
    \[ k_i = \min\{ k \in \N: \dim\Span_\R\{ x_1, \dots, x_k\} = i\} \text{, set } v_i=x_{k_i} \text{ and let } \lambda_i = \|x_i\|_C.\]
    Clearly, $v_i \in \lambda_iC \cap \Lambda$ and $0 < \lambda_1 \leq \dots \leq \lambda_n < \infty$. {\color{blue} Take a basis $\{ v_1, \dots, v_n\} \subseteq \Lambda$ of $\R^n$. We have $\max_{i \in \{ 1, \dots, n\}}|v_i| \in \R$. There are only finitely many vectors with a smaller norm than this. Hence, the finiteness claim holds.} Finally, we show that these are indeed the successive minima: By construction, the
    \[ v_1, \dots, v_i \in \lambda_iC \cap \Lambda \text{ are linearly independent over } \R.\]
    Take $0 < \lambda < \lambda_i$ and let $\ell$ be the largest number such that $x_{\ell} \in \lambda C \cap \Lambda$. Then, we have $\ell < k_i$, and by the definition of the $k_i$, it follows that
    \[ \dim\Span_\R(\lambda C \cap \Lambda) = \dim\Span_\R\{ x_1, \dots, x_{\ell}\} < i.\]
    This shows that $\lambda_i$ is indeed the successive minimum.
    \end{pf}
    \begin{theorem}{Minkowski's Second Convex Body Theorem}{2.9}
      Let $\Lambda \subseteq \R^n$ be a lattice and $C \subseteq \R^n$ a central symmetric convex body. Further assume that $\lambda_1, \dots, \lambda_n$ are the successive minima of $\Lambda$ with respect to $C$. Then, we have
      \[ \frac{2^n}{n!}\frac{\det\Lambda}{\vol(C)} \leq \lambda_1\cdots\lambda_n \leq 2^n\frac{\det\Lambda}{\vol(C)}.\]
    \end{theorem}
    \noindent Before we prove this, let us study a couple of examples to highlight the importance of this result.
    \begin{example}
      Let $n \geq 2$ and
      \[ \mathfrak B = B_1^n(0).\]
      Further let $\Lambda \subseteq \R^n$ be a lattice and $\lambda_1, \dots, \lambda_n$ be the successive minima of $\Lambda$ with respect to $\mathfrak B$.       \[ \prod_{i=1}^n |x_i| \leq \frac{2^n\det\Lambda}{V_n},\]
      where
      \[ V_n = \vol(\mathfrak B) = \begin{cases}
        1 & \text{if } n=1, \\
        \pi & \text{if } n=2, \\
        \frac{2\pi}{n}V_{n-2} & \text{if } n \geq 3. \hspace{8.8cm} \qedhere
      \end{cases}\]
    \end{example}

      Let $\Lambda = \Z^n$. Take any real $0 < \lambda_1 < \dots < \lambda_n$. Define $C = \{ x \in \R^n: |x_i| \leq \nicefrac{1}{\lambda_i} \forall_{1 \leq i \leq n}\}$. Then $C$ is a central symmetric convex body with

      \[ \lambda C = \{ x \in \R^n: |x_i| \leq \nicefrac{\lambda}{\lambda_i} \forall_{1 \leq i \leq n}\}.\]

        = \quad & \|(y, z) + (\alpha - 1) (y_0, z)\|_C \notag\\
        \le \quad & \|(y, z)\|_C + (\alpha - 1) \|(y_0, z)\|_C \notag\\
        \le  \quad & t + (\alpha - 1) t \notag\\
        = \quad & \alpha t. \tag{3}
        \label{eqn:minkowski2:3}
    \end{align}
    Let us define a notation: given \(y \in \R^j\), let
    \begin{equation*}
        \{y\}
        = (y_1 - \lf y_1\rf, \dots, y_j - \lf y_j\rf) \in [0,1)^j.
    \end{equation*}
    Then
    \begin{equation*}
        S_j(t, z)
        = \{\{y \}\in [0,1)^j \mid y \in \R^j, \|(y, z)\|_C \le t\}.
    \end{equation*}
    Now, we define
    \begin{equation*}
        W
        = \{\{y + (\alpha - 1) y_0\} \in [0,1)^j \mid y \in \R^j, \|(y, z)\|_C \le t\}.
    \end{equation*}
    Then \(\vol(S_j(t, z)) = \vol(W)\) (by the translation invariance of the Lebesgue measure).
    Furthermore, by~\cref{eqn:minkowski2:3}, we have
    \begin{align*}
        W
        \subseteq \quad & \{\{y + (\alpha - 1) y_0\} \in [0,1)^n \mid y \in \R^j, \|(y + (\alpha - 1) y_0, \alpha z)\|_C \le \alpha t\} \\
        = \quad & \{\{y\} \in [0,1)^j \mid y \in \R^j, \|(y, \alpha z)\|_C \le \alpha t\} \\
        = \quad & S_j(\alpha t, \alpha z).
    \end{align*}
    Take the volume to obtain
    \begin{equation*}
        \vol(S_j(t, z))
        = \vol(W)
        \le \vol(S_j(\alpha t, \alpha z)),
    \end{equation*}
    proving the claim.

    Suppose that \(0 < t \le \alpha t < \lambda_{j+1} / 2\).
    Then by \cref{eq:mink_2} and the claim,
    \begin{align*}
        \vol(S(\alpha t))
        = \quad & \int_{\R^{n-j}} \vol(S_j(\alpha t, z)) \dl{z} \\
        = \quad & \alpha^{n-j} \int_{\R^{n-j}} \vol(S_j(\alpha t, \alpha z)) \dl{z} \\
        \ge \quad & \alpha^{n-j} \int_{\R^{n-j}} \vol(S_j(t, z)) \dl{z}.
    \end{align*}
    This gives
    \begin{equation}
        \vol(S(\alpha t))
        \ge \alpha^{n-j} \vol(S(t)). \tag{4}
        \label{eqn:minkowski2:4}
    \end{equation}

    We are ready to prove the theorem.
    When \(t < \lambda_1 / 2\), we have
    \begin{equation*}
        \vol(S(t))
        = \vol(tC)
        = t^n \vol(C).
    \end{equation*}
    This is because if \(\eta + m, \eta + m' \in tC\) with \(m, m' \in \Z^n\), then
    \begin{equation*}
        \|m - m'\|_C
        = \|(\eta + m) - (\eta + m')\|_C
        \le \|\eta + m\|_C + \|\eta + m'\|_C
        \le 2t
        < \lambda_1,
    \end{equation*}
    so \(m = m'\).
    Therefore, \(\vol(S(t)) = \vol(tC) = t^n \vol(C)\) if \(t < \lambda_1 / 2\).
    Since \(S(t) \subseteq S(\lambda_1 / 2)\) for any \(t \le \lambda_1 / 2\), we have \(\vol(S(\lambda_1 / 2)) \ge (\lambda_1 / 2)^n \vol(C)\).

    We now use induction to prove the statement for \(1 \le j \le n-1\).
    Suppose that the statement holds for any \(t \le \lambda_j / 2\) for some~\(j\).
    Consider \(\lambda_j / 2 \le t < \lambda_{j+1} / 2\).
    Set \(t_0 = \lambda_j / 2\) and \(\alpha \in [1, \lambda_{j+1} / \lambda_j)\) such that \(t = \alpha t_0\).
    Then
    \begin{align*}
        \vol(S(t))
        = \quad & \vol(S(\alpha t_0)) \\
        \ge \quad & \alpha^{n-j} \vol(S(t_0)) \\
        \ge \quad & \alpha^{n-j} \left(\frac{\lambda_1}{2} \dotsm \frac{\lambda_{j-1}}{2}\right) t_0^{n-j+1} \vol(C) \\
        = \quad & t^{n-j} \left(\frac{\lambda_1}{2} \dotsm \frac{\lambda_{j-1}}{2} \frac{\lambda_j}{2}\right) \vol(C).
    \end{align*}
    And for \(t = \lambda_{j+1} / 2\), this holds by taking the limit since \(S(t) \subseteq S(\lambda_{j+1} / 2)\) if \(t \le \lambda_{j+1} / 2\).
    This completes induction and gives us the statement for all \(t \le \lambda_n / 2\).

    Finally, suppose that \(t \ge \lambda_n / 2\).
    Then \(S(\lambda_n / 2) \subseteq S(t)\), and so
    \begin{align*}
        \vol(S(t))
        \ge \quad & \vol(S(\lambda_n / 2)) \\
        \ge \quad & \frac{\lambda_1}{2} \dotsm \frac{\lambda_n}{2} \vol(C).
    \end{align*}
    (This follows from the case \(t = \lambda_n / 2\)\@.)
\end{pf}
    \noindent We shall now prove Minkowski's Second Theorem:

\section{Dual lattice}

Let \(\Lambda\) be a lattice in~\(\R^n\).
We define the dual lattice~\(\Lambda^*\) to be
\begin{equation*}
    \Lambda^*
    = \{x \in \R^n \mid \langle x,y\rangle \in \Z \forall_{y \in \Lambda}\}.
\end{equation*}
Let \(\{v_1, \dots, v_n\}\) be a basis of~\(\Lambda\) and let \(A = \begin{pmatrix} v_1 & \cdots & v_n\end{pmatrix} \).
Then
\begin{equation*}
    \begin{split}
        \Lambda^*
        &= \{x \in \R^n \mid \inp{x}{y} \in \Z \forall_{y \in \Lambda}\} \\
        &= \{x \in \R^n \mid \inp{x}{A z} \in \Z\forall_{z \in \Z^n}\} \\
        &= \{x \in \R^n \mid \inp{A^t x}{z} \in \Z\forall_{z \in \Z^n}\} \\
        &= \{x \in \R^n \mid A^t x \in \Z^n\} \\
        &= \{(A^t)^{-1} w \mid w \in \Z^n\}.
    \end{split}
\end{equation*}
Hence,
\begin{equation*}
    \det \Lambda^*
    = |\det(A^t)^{-1}|
    = \frac{1}{\abs{\det A}}
    = \frac{1}{\det \Lambda}.
\end{equation*}

\begin{theorem}{Estimate for Products of Successive Minima and Dual Successive Minima}{2.13}
    Let \(\Lambda\) be a lattice in~\(\R^n\) and let \(\lambda_1, \dots, \lambda_n\) be its successive minima with respect to \(B = \{x \in \R^n \mid \|x\|_2 \le 1\}\).
    Let \(\Lambda^*\) be its dual, and let \(\lambda_1^*, \dots, \lambda_n^*\) be the successive minima of~\(\Lambda^*\) with respect to \(B\).
    Then
    \begin{equation*}
        1
        \le \lambda_i \lambda_{n-i+1}^*
        \le c(n)
    \end{equation*}
    for all \(1 \le i \le n\) where \(c(n)\) is a constant that depends only on~\(n\).
\end{theorem}

\noindent Successive minima for dual lattices are only considered with respect to the unit ball.

\begin{pf}
    We begin with the lower bound.
    Let \(v_1, \dots, v_n \in \Lambda\) be linearly independent over~\(\R\) and \(\|v_i\|_2 = \lambda_i\) for all \(1 \le i \le n\) (from Lemma \ref{lem:2.8}).
    Similarly, take \(v_1^*, \dots, v_n^* \in \Lambda^*\) linearly independent over~\(\R\) with \(\|v_i^*\|_2 = \lambda_i^*\) for all \(1 \le i \le n\).
    Take \(1 \le i \le n\).
    Define
    \begin{equation*}
        V_i
        = \{x \in \R^n \mid \inp{x}{v_k} = 0 \text{ for each } 1 \le k \le i\}.
    \end{equation*}
    Since \(v_1, \dots, v_i\) are linearly independent over~\(\R\), we obtain \(\dim V_i = n-i\).
    Therefore, at least one of \(v_1^*, \dots, v_{n+1-i}^*\) is not in~\(V_i\).
    Thus, there is \(1 \le i_0 \le i\) and \(1 \le k_0 \le n+1-i\) such that \(\inp{v_{i_0}}{v_{k_0}^*} \ne 0\).
    From the definition of~\(\Lambda^*\), we have \(\abs{\inp{v_{i_0}}{v_{i_0}^*}} \in \Z\) and
    \begin{equation*}
        1
        \le \abs{\inp{v_{i_0}}{v_{i_0}^*}}
        \stackrel{\text{C.-S. I.}}{\le} \|v_{i_0}\|_2 \cdot \|v_{k_0}^*\|_2
        = \lambda_{i_0} \lambda_{k_0}^*
        \leq \lambda_i \lambda_{n+1-i}^*.
    \end{equation*}

    Next, the upper bound.
    By Theorem \ref{thm:2.9}, we have
    \begin{equation*}
        \lambda_1 \dotsm \lambda_n
        \le \frac{2^n}{\vol(B)} \det \Lambda, \qquad
        \lambda_1^* \dotsm \lambda_n^*
        \le \frac{2^n}{\vol(B)} \det \Lambda^*.
    \end{equation*}
    This gives
    \begin{equation*}
        \prod_{i=1}^n \lambda_i \lambda_{n+1-i}^*
        \leq \frac{4^n}{\vol(B)^2} \det \Lambda \det \Lambda^*
        = \frac{4^n}{\vol(B)^2}.
    \end{equation*}
    Take any \(1 \le i_0 \le n\).
    Then
    \begin{equation*}
        \lambda_{i_0} \lambda_{n+1-i_0}^*
        = \frac{4^n}{\vol(B)^2} \frac{1}{\prod_{i \ne i_0} \underbrace{\lambda_i \lambda_{n+1-i}^*}_{\ge 1}}
        \le \frac{4^n}{\vol(B)^2}. \qedhere
    \end{equation*}
\end{pf}

\noindent As an application, we will show that there is no \glqq small\grqq \ vector in~\(\Lambda^*\).
This gives a good approximation of vectors in~\(\R^n\) by~\(\Lambda\).

\begin{corollary}{Approximation of Vectors in $\R^n$ with Lattice Vectors}{2.14}
    Let us use the same notation as in Theorem \ref{thm:2.13}.
    Suppose that \(0 < R \le \lambda_1^*\).
    Then, given any \(b \in \R^n\),
    \begin{equation*}
        \exists_{x \in \Lambda}: \|b - x\|_2
        \le \frac{n c(n)}{2R}.
    \end{equation*}
\end{corollary}

\begin{pf}
    Since \(\lambda_1^* \ge R\), by Theorem \ref{thm:2.13}, we obtain
    \begin{equation*}
        \lambda_n
        \le \frac{c(n)}{\lambda_1^*}
        \le \frac{c(n)}{R}.
    \end{equation*}
    Let \(v_1, \dots, v_n \in \Lambda\) be linearly independent over~\(\R\) with \(\|v_i\|_2 = \lambda_i \le \lambda_n \le c(n) / R\) for all \(1 \le i \le n\).
    Take any \(b \in \R^n\), and write \(b = \xi_1 v_1 + \dots + \xi_n v_n\).
    Let \(m_1, \dots, m_n \in \Z\) such that \(\abs{\xi_i - m_i} \le \frac{1}{2}\).
    Then \(x = m_1 v_1 + \dots + m_n v_n \in \Lambda\).
    We have
    \begin{equation*}
        \|x - b\|_2
        = \|(\xi_1 - m_1) v_1 + \dots + (\xi_n - m_n) v_n\|_2
        \le \frac{1}{2} (\|v_1\|_2 + \dots + \|v_n\|_2)
        \le \frac{n}{2} \frac{c(n)}{R}. \qedhere
    \end{equation*}
\end{pf}

\begin{theorem}{Shifted Simultaneous Diophantine Approximation}{2.15}
  Let $\alpha_1, \dots, \alpha_n, \theta_1, \dots, \theta_n \in \R$. Suppose that $\{ 1, \alpha_1, \dots, \alpha_n\}$ is linearly independent over $\Q$. Then, for any constant $\epsilon > 0$, there exist infinitely many $(x_1, \dots, x_n, y) \in \Z^{n+1}$ such that
  \[ |\alpha_iy - x_i-\theta_i| \leq \epsilon \forall_{1 \leq i \leq n}.\]
\end{theorem}
\noindent \textbf{Remark:} Recall Corollary \ref{cor:2.7}, which was a multi-dimensional Dirichlet's theorem. This is a version with a shift $\theta$ (inhomogeneous), and a nice application of the concept of a dual lattice. Also note that the condition of the linear independence is necessary (exercise).
\begin{pf}
  Let $M$ be a sufficiently large integer. Also let
  \[ A = \begin{pmatrix}
    & & & -\alpha_1 \\
    & \1_n & & \vdots \\
    & & & -\alpha_n \\
    0 & \cdots & 0 & M^{-1}
  \end{pmatrix}\]
  and
  \begin{align*} \Lambda_M = \quad & \{ Az: z = (x_1, \dots, x_n,y) \in \Z^{n+1}\} \\
  = \quad & \{ (x_1-\alpha_1y, \dots, x_n-\alpha_ny, M^{-1}y): (x_1, \dots, x_n, y) \in \Z^{n+1}\}.\end{align*}
  Write
  \[ b = \begin{pmatrix}
    \theta_1 \\
    \vdots \\
    \theta_n \\
    2\epsilon
  \end{pmatrix}.\]
  The goal is to show that \[\exists_{z \in \Z^{n+1}}: \|b-Az\| \leq \epsilon. \hfill (*)\]
  This would imply
  \begin{align*} \epsilon \geq \quad & \sqrt{(x_1-\alpha_1y-\theta_1)^2 + \dots + (x_n-a_ny-\theta_n)^2 + (M^{-1}y-2\epsilon)^2} \\
   \geq \quad & \begin{cases}
    |x_i-\alpha_iy-\theta_i| & 1 \leq i \leq n, \\
    |M^{-1}y-2\epsilon| & i=n+1.
  \end{cases}\end{align*}
  The latter is equivalent to $\epsilon M \leq y \leq 3\epsilon M$. Hence, we can generate infinitely many $(x_1, \dots, x_n, y) \in \Z^{n+1}$ that satisfy the statement of the theorem. We have
  \[ (A^t)^{-1} = \begin{pmatrix}
    & & & 0 \\
    & \1_n & & \vdots \\
    & & & 0 \\
    M\alpha_1 & \dots & M\alpha_n & M
  \end{pmatrix}.\]
  Therefore, our dual lattice is
  \[ \Lambda_M^* = (A^t)^{-1}\Z^{n+1} = \{ (x_1, \dots, x_n, M(\alpha_1x_1+\dots+\alpha_nx_n+y)): x_1, \dots, x_n,y \in \Z\}.\]
  Let
  \[ R = \frac{(n+1)C(n+1)}{2\epsilon}\]
  and
  \[ \mu = \min_{(x_1, \dots, x_n,y) \in S} |\alpha_1x_1+\dots+\alpha_nx_n+y|,\]
  where
  \[ S = \{ (x_1, \dots, x_n,y) \in \Z^{n+1} \setminus \{ 0\} : |x_i| < R \forall_{1 \leq i \leq n} \land |\alpha_1x_1+\dots+\alpha_nx_n+y| \leq 1\}.\]
  The set $S$ is clearly finite. Recall that $\alpha_1, \dots, \alpha_n$ are linearly independent over $\Q$. Hence, $\mu > 0$. We choose
  \[ M > \max\left\{ \frac{R}{\mu}, R\right\}.\]
  Then, given any $u = (x_1, \dots, x_n, y) \in \Lambda_M^* \setminus \{ 0\}$, we have
  \[ \|u\|_2 = \sqrt{x_1^2+\dots+x_n^2+M^2(\alpha_1x_1+\dots+\alpha_nx_n+y)^2} \geq R,\]
  which we can prove by considering the following three cases:
  \begin{enumerate}[(i)]
    \item $\exists_{1 \leq i \leq n}: |x_i| \geq R. \ \checkmark$
    \item $\forall_{1 \leq i \leq n}: |x_i| < R, (x_1, \dots, x_n,y) \notin S$ \\
    Use $M > R$.
    \item $\forall_{1 \leq i \leq n}: |x_i| < R, (x_1, \dots, x_n,y) \in S$ \\
    Use $M > \nicefrac{R}{\mu}$.
  \end{enumerate}
  This means that $\lambda_1^* > R$ (first minimum of $\Lambda_M^*$ with respect to the unit ball). We can then apply Corollary \ref{cor:2.14} to deduce that
  \[ \exists_{v = Az \in \Lambda_M^*}: \|b-Az\|_2 \leq \frac{(n+1)C(n+1)}{2R} = \epsilon.\]
  This is the bound $(*)$, which finishes the proof.
\end{pf}
\section{Lagrange's Four Square Theorem}
\begin{theorem}{Lagrange's Four Square Theorem, 1770}{2.16}
  Given $n \in \N$, $\exists_{a,b,c,d \in \Nzero}$ such that
  \[ n = a^2+b^2+c^2+d^2.\]
\end{theorem}
\noindent We'll do a Geometry of Numbers-style proof of this. We begin with a simple lemma:
\begin{lemma}{Two Square Residue Lemma}{2.17}
  Given any $p \in \PP$, there exist $a,b \in \Z$ such that
  \[ a^2+b^2 \equiv -1 \mod p.\]
\end{lemma}
\begin{pf}
  If $p=2$, then $(a,b)=(1,0)$ works. For $p > 2$, define
  \[ \A = \{ a^2 \smod p: 0 \leq a < \nicefrac{p}{2}\}.\]
  The elements of $\A$ are distinct because
  \[ a^2 \equiv \widehat{a}^2 \mod p \implies p \mid a^2-\widehat{a}^2 = (a+\widehat{a})(a-\widehat{a}),\]
  from which we can immediately deduce that $p \mid a-\widehat{a}$ {\color{blue}, possibly by $a = \widehat{a} = 0$}, id est $a \equiv \widehat{a} \mod p$. Therefore,
  \[ \#\A = \frac{p-1}{2}+1 = \frac{p+1}{2}.\]
  Similarly, we define
  \[ \mathfrak{B} = \{ -1-b^2 \smod p: 0 \leq b < \nicefrac{p}{2}\}.\]
  By the same argument,
  \[ \#\mathfrak B = \frac{p+1}{2}.\]
  Therefore, we can use the pigeonhole principle to deduce that $\A \cap \mathfrak B \neq \varnothing$, which means that
  \[ \exists_{a \in \A, b \in \mathfrak B}: a^2 \equiv -1-b^2 \mod p. \qedhere\]
\end{pf}
\noindent We also need the following lemma, the proof of which will be on the next exercise sheet.
\begin{lemma}{Lattice Generated by Linear Forms}{2.18}
  Let $k_1, \dots, k_m \in \N$, $m \geq 1$. Assume that $L_1(x), \dots, L_m(x) \in \Z[x]$ are linear forms. Define
  \[ \Lambda = \{ x \in \Z^n: L_j(x) \equiv 0 \smod k_j \forall_{1 \leq j \leq m}\}.\]
  Then $\Lambda$ is a lattice with $\det\Lambda \leq k_1\cdots k_m$.
\end{lemma}
\begin{pf}
  Exercise.
\end{pf}
\noindent These two lemmata allow us to prove Theorem \ref{thm:2.16}.
\begin{pf}[Proof of Theorem \ref{thm:2.16}]
  For $n=1$, the statement is trivial. Otherwise, write $n = m^2p_1\cdots p_r$, where $p_1 \neq \cdots \neq p_r \in \PP$. Writing $\widetilde{a} = ma, \dots, \widetilde{d} = md$, it suffices to consider $n=p_1\cdots p_r$. We define
  \[ \{ (u_1, \dots, u_4) \in \Z^4: u_1-a_pu_3-b_pu_4 \equiv 0 \smod p, u_2-b_pu_3+a_pu_4 \equiv 0 \smod p \forall_{\PP \ni p \mid n}\},\]
  where $a_p, b_p$ are such that $a_p^2+b_p^2 \equiv -1 \mod p$ (use Lemma \ref{lem:2.17}). Then $\Lambda$ is a lattice with $\det\Lambda \leq n^2$. Let $\epsilon > 0$ be sufficiently small with respect to $n$. Define
  \[ C = B^4_{\sqrt{2(n-\epsilon)}}(0).\]
  Then
  \[ \vol(C) = \frac{\pi^2}{2}\sqrt{2(n-\epsilon)}^4 =2\pi^2(n-\epsilon)^2 >  16n^2 \geq 2^4\det\Lambda.\]
  Now, Minkowski's First Theorem \ref{thm:mink_first_conv_bod_thm} implies that there exists $0 \neq u \in \Lambda \cap C$. In particular,
  \[ 0 < u_1^2+u_2^2+u_3^2+u_4^2 \leq 2(n-\epsilon) <2n.\]
  We also have $u \in \Lambda$, which means that
  \begin{align*} u_1^2+u_2^2+u_3^2+u_4^2 \equiv \quad &  (a_pu_3+b_pu_4)^2+(b_pu_3-a_pu_4)^2+u_3^2+u_4^2 \\
  \equiv \quad & (a_p^2+b_p^2+1)u_3^2 + (a_p^2+b_p^2+1)u_4^2 \\
  \equiv \quad & 0 \hspace{9cm} \mod p\end{align*}
  for each $\PP \ni p \mid n$. \\
  Hence, $p \mid u_1^2+u_2^2+u_3^2+u_4^2 \forall_{\PP \ni p \mid n}$, which implies $n \mid u_1^2+u_2^2+u_3^2+u_4^2$. Combining this with the upper bound on $n$ yields
  \[ n = u_1^2+u_2^2+u_3^2+u_4^2. \qedhere\]
\end{pf}
\section{Short Vectors in a Lattice}
Apparently, finding the shortest vector in a lattice is an important (and quite difficult) problem in cryptography. Some cryptographic schemes rely on this fact. The secret corresponds to the shortest vector. Though we cannot do this in general, we can find a \glqq pretty short\grqq \ vector (LLL-Algorithm). We will do this next week.

\subsection{Reduced Basis}
Let $\{ b_1, \dots, b_n \}$ be linearly independent vectors in $\R^n$.
The Gram--Schmidt orthogonalisation process produces vectors
$\{ b_1', \dots, b_n' \}$ defined recursively by
\[
b_1' = b_1,
\]
\[
b_2' = b_2 - \mu_{2,1}b_1',
\]
\[
\quad \  \vdots
\]
\[
b_i'
=
b_i
-
\sum_{j=1}^{i-1}\mu_{i,j}b_j',
\]
\[
\quad \ \vdots
\]
\[
b_n'
=
b_n
-
\sum_{j=1}^{n-1}\mu_{n,j}b_j',
\]
where
\[
\mu_{i,j}
=
\frac{\inp{b_i}{b_j'}}
     {\inp{b_j'}{b_j'}}
\forall_{1 \leq i \leq n,\; 1 \leq j \leq i-1}.
\]

\noindent These vectors satisfy
\[
\inp{b_i'}{b_j'} = 0
\qquad \text{for } i \neq j.
\]
\para{Recall:}
We can write the vector $b_i$ as
\[ b_i = b_i' + \sum_{j=1}^{i-1} \mu_{i,j} b_j', \]
where $\displaystyle \sum_{j=1}^{i-1} \mu_{i,j} b_j'$ is the projection of $b_i$ onto $\Span_{\mathbb{R}} \{ b_1, \dots, b_{i-1} \} = \Span_\R\{ b_1', \dots, b_{i-1}'\}$.

\begin{definition}{Reduced Basis}{2.18}
    A basis $\{ b_1, \dots, b_n \}$ of a lattice $\Lambda \subseteq \mathbb{R}^n$ is \define[Reduced Basis of a Lattice]{reduced} if
    \begin{enumerate}[(i)]
        \item $|\mu_{i,j}| \le \frac{1}{2}$ for $1 \le j < i \le n$ and
        \item $\|b_i' + \mu_{i,i-1} b_{i-1}'\|_2^2 \ge \frac{3}{4} \|b_{i-1}'\|_2^2$ for $2 \le i \le n$.
    \end{enumerate}
\end{definition}

\titremark{The constant $3/4$} The choice of the constant $\frac{3}{4}$ is not crucial; any real number in the interval $(\frac{1}{4}, 1)$ works. Also,
\[ b_i = {\color{red} b_i' + \mu_{i,i-1}b_{i-1}'} + \sum_{j=1}^{i-2}\mu_{i,j}b_j',\]
so condition (ii) essentially says that the highlighted portion of $b_i$ is longer (up to a constant factor) than $b_{i-1}'$.

\begin{proposition}{Properties of Reduced Bases}{2.19}
    Let $\{ b_1, \dots, b_n \}$ be a reduced basis of $\Lambda$. Let $\{ b_1', \dots, b_n' \}$ be the Gram-Schmidt orthogonalization. Then
    \begin{enumerate}[(i)]
        \item $\|b_l\|_2^2 \le 2^{i-1} \|b_i'\|_2^2$ for $1 \le l \le i \le n$,
        \item $\det \Lambda \le \|b_1\|_2 \cdots \|b_n\|_2 \le 2^{\frac{n(n-1)}{4}} \det \Lambda$ and
        \item $\|b_1\|_2 \le 2^{\frac{n-1}{4}} (\det \Lambda)^{1/n}$.
    \end{enumerate}
\end{proposition}

\begin{pf} \hfill
    \begin{enumerate}[(i)] \item This follows from the definition. The statement is trivial for $i=1$.
    For $2 \le i \le n$, condition (ii) in Definition \ref{def:2.18} implies:
    \[ \|b_i'\|_2^2 + \mu_{i,i-1}^2 \|b_{i-1}'\|_2^2 \ge \frac{3}{4} \|b_{i-1}'\|_2^2 \]
    Since $|\mu_{i,i-1}| \le \frac{1}{2}$, we have $\mu_{i,i-1}^2 \le \frac{1}{4}$, which gives:
    \[ \|b_i'\|_2^2 \ge \left( \frac{3}{4} - \frac{1}{4} \right) \|b_{i-1}'\|_2^2 = \frac{1}{2} \|b_{i-1}'\|_2^2 \]
    By induction, we have \[2^{i-l} \|b_i'\|_2^2 \ge \|b_l'\|_2^2\text{ for }1 \le l \le i. \hfill (\star)\]
    Now, relating $\|b_l\|_2$ to $\|b_l'\|_2$:
    \[ \|b_l\|_2^2 = \|b_l'\|_2^2 + \sum_{j=1}^{l-1} \mu_{l,j}^2 \|b_j'\|_2^2 \le \|b_l'\|_2^2 + \sum_{j=1}^{l-1} \frac{1}{4} \cdot 2^{l-j} \|b_l'\|_2^2 \]
    Summing the geometric series yields $\|b_l\|_2^2 \le 2^{l-1} \|b_l'\|_2^2$. Combined with ($\star$), we get the result.

    \item The lower bound is Hadamard's inequality. For the upper bound, we use $\|b_i\|_2 \le 2^{\frac{i-1}{2}} \|b_i'\|_2$ and the fact that \[\det \Lambda = \prod_{i \in \{ 1, \dots, n\}} \|b_i'\|_2.\]
    \item Applying (i) with $l=1$ gives $\|b_1\|_2^2 \le 2^{i-1} \|b_i'\|_2^2$. Taking the product over $i=1, \dots, n$ and then the $2n$-th root leads to the bound. \qedhere
    \end{enumerate}
\end{pf}

\begin{proposition}{Shortest Vector Approximation}{2.20}
    Let $\{ b_1, \dots, b_n \}$ be a reduced basis of $\Lambda$. For any $0 \neq x \in \Lambda$:
    \[ \|b_1\|_2^2 \le 2^{n-1} \|x\|_2^2 \]
\end{proposition}

\noindent \textbf{Remark:} This implies that $b_1$ is within a factor of $2^{\frac{n-1}{2}}$ of the shortest vector. The LLL algorithm computes such a basis efficiently. In modern cryptography, $n$ often ranges from 256 to over 1024.

\begin{pf}
Write
\begin{align*} x = \quad & m_{1}b_{1} + \dots + m_{n}b_{n} \\
 = \quad & \alpha_{1}b_{1}^{\prime} + \dots + \alpha_{n}b_{n}^{\prime} \end{align*}
 with $m_1, \dots, m_n \in \Z$ and $\alpha_{1}, \dots, \alpha_{n} \in \mathbb{R}$.
Let $j_{0}$ be the largest index such that
\[ m_{j_{0}} \neq 0. \]

Then
\[ \Span_{\mathbb{R}}\{b_{1}, \dots, b_{j_{0}}\} = \Span_{\mathbb{R}}\{b_{1}^{\prime}, \dots, b_{j_{0}}^{\prime}\} \]
and it follows from the Definition of $b_{1}^{\prime}, \dots, b_{n}^{\prime}$ that $m_{j_{0}} = \alpha_{j_{0}}$. We further have
\[ \alpha_{j_{0}+1} = \dots = \alpha_{n} = 0. \]

Then
\begin{align*}
\|x\|_{2}^{2} &\ge \alpha_{j_{0}}^{2} \|b_{j_{0}}^{\prime}\|_{2}^{2} \\
&\ge \|b_{j_{0}}^{\prime}\|_{2}^{2} \\
&\ge 2^{-(j_{0}-1)} \cdot \|b_{1}\|_{2}^{2}
\end{align*}
by Proposition \ref{prop:2.19}(i).
This implies
\[ \|b_{1}\|_{2} \le 2^{\frac{j_{0}-1}{2}} \|x\|_{2} \le 2^{\frac{n-1}{2}} \|x\|_{2}. \qedhere \]
\end{pf}

\begin{proposition}{Generalised Shortest Vector Approximation}{2.21}
Let $\{b_{1}, \dots, b_{n}\}$ be a reduced basis of a lattice $\Lambda$. Let $1 \le \ell \le n$ and $x_{1}, \dots, x_{\ell} \in \Lambda$ be linearly independent over $\mathbb{R}$.

Then
\[ \|b_{i}\|_{2}^{2} \le 2^{n-1} \cdot \max \{ \|x_{1}\|_{2}^{2}, \dots, \|x_{\ell}\|_{2}^{2} \} \]
for all $1 \le i \le \ell$.
\end{proposition}

\begin{pf}
For each $x_{i}$, write
\begin{align*} x_{i} = \quad & m_{i,1}b_{1} + \dots + m_{i,n}b_{n} \\
 = \quad & \alpha_{i,1}b_{1}^{\prime} + \dots + \alpha_{i,n}b_{n}^{\prime} \end{align*}
with $m_{i,j} \in \mathbb{Z}$ and $\alpha_{i,j} \in \mathbb{R}.$
Let $j_{0}(i)$ be the largest $j$ such that $m_{i,j} \neq 0$.

Then, by the definition of $\{b_{1}^{\prime}, \dots, b_{n}^{\prime}\}$, it follows that
\[ \alpha_{i, j_{0}(i)+1} = \dots = \alpha_{i,n} = 0 \]
and
\[ 0 \neq m_{i, j_{0}(i)} = \alpha_{i, j_{0}(i)},\]
as before.

We have

Without loss of generality (relabel $x_{1}, \dots, x_{\ell}$, if necessary) we assume
\[ j_{0}(1) \le \dots \le j_{0}(\ell). \]

\begin{claim*}{} For each $1 \le i \le \ell$, we have $i \le j_{0}(i)$.
\end{claim*}

Suppose otherwise. Then $j_{0}(i) < i$, but

The left side is, however, linearly independent over $\mathbb{R}$. \Lightning \\

\section{The LLL Algorithm}

The \textbf{LLL (Lenstra-Lenstra-Lovász) algorithm} turns a basis into a reduced basis (and thus gives a \glqq short\grqq \ vector in $\Lambda$, recall Proposition \ref{prop:2.20}).

\begin{definition}{Property($k$)}{prop_k}
    Let $2 \le k \le n+1$. We say a basis $\{ b_1, \dots, b_n \}$ satisfies \define[$k$-th Property for Bases of Lattices]{Property($k$)} if:
    \begin{enumerate}[(i)]
        \item $|\mu_{i,j}| \le \frac{1}{2}$ for all $1 \le j < i < k$.
        \item $\|b_i' + \mu_{i,i-1}b_{i-1}'\|_2^2 \ge \frac{3}{4} \|b_{i-1}'\|_2^2$ for all $2 \le i < k$.
    \end{enumerate}
\end{definition}

\noindent \textbf{Note:} Property$(n+1)$ means the basis is reduced. And Property$(2)$ holds trivially for any basis.

\subsection{How the Algorithm Works}

We start with a basis $B_0 = \{ b_1, \dots, b_n \}$ of $\Lambda$. In particular, it satisfies Property$(2)$.
The algorithm proceeds in steps, producing a sequence of bases:
$$ B_0 \xrightarrow{\text{Prop}(2)} B_1 \xrightarrow{\text{Prop}(k_1)} B_2 \xrightarrow{\text{Prop}(k_2)} \dots \to B_{\text{final}} \xrightarrow{\text{Prop}(n+1)} $$
where $k_{i+1} = k_i \pm 1$. Once we reach Property$(n+1)$, we get a reduced basis. Done! As we shall see, $k$ can go down only a finite number of times, so the algorithm terminates.

\titremark{Efficiency of the Algorithm} The algorithm is efficient. If we start with a basis $\{ b_1, \dots, b_n \}$ with $\|b_i\|_2 \le B$ ($1 \le i \le n$), then the number of arithmetical operations (addition, subtraction, multiplication, division) needed in the algorithm is $\mathcal{O}(n^4 \log B)$, and the algorithm runs in polynomial time in $B$.

\subsection{Steps of the Algorithm}
\paragraph{Initial step:} Let $2 \le k \le n$. Assume we have a partial basis satisfying Property$(k)$. We first achieve
\[ |\mu_{k,k-1}| \le \frac{1}{2}. \hfill (*) \]
If $(*)$ does not hold, let $r \in \mathbb{Z}$ be such that $|\mu_{k,k-1} - r| \le \frac{1}{2}$. We replace $b_k$ with $b_k - rb_{k-1}$. This has the effect of changing $\mu_{k,k-1}$ to $\mu_{k,k-1} - r$. Since $|\mu_{k,k-1} - r| \le \frac{1}{2}$, the new basis satisfies $(*)$.

\noindent Now we have a basis $\{ b_1, \dots, b_n \}$ satisfying Property$(k)$ and $(*)$. We distinguish two main cases:

\paragraph{Case 1 (Swap Step):} If $k \ge 2$ and
\[ \|b_k' + \mu_{k,k-1}b_{k-1}'\|_2^2 < \frac{3}{4} \|b_{k-1}'\|_2^2 \hfill (\widehat{*}) \]
Then we switch $b_k$ and $b_{k-1}$ while keeping the rest the same. Our new basis is $c_1, \dots, c_n$ with $c_{k-1} = b_k$, $c_k = b_{k-1}$, and $c_i = b_i$ otherwise.
This changes $\mu_{k-1,j}$ and $\mu_{k,j}$, but it does not affect $b_i, b_i', \mu_{i,j}$ for $1 \le i \le k-2$. The new basis still satisfies Property$(k-1)$.

\noindent Now $c_{k-1}'$ is the projection of $c_{k-1} = b_k$ onto $(\Span_{\mathbb{R}}\{b_1', \dots, b_{k-2}'\})^\perp$, i.e.
\[ c_{k-1}' = b_k' + \mu_{k,k-1}b_{k-1}'. \]
From $(\widehat{*})$, we get $\|c_{k-1}'\|_2^2 < \frac{3}{4} \|b_{k-1}'\|_2^2$. So the length of the \glqq new\grqq \ $c_{k-1}'$ is shorter than the \glqq old\grqq \ $b_{k-1}'$ by a factor of $3/4$.
\titremark{Note} This is what is used to show that $k$ can go down only finitely many times.

\paragraph{Case 2 (Size Reduction Step):} If $k \ge 2$ and
\[ \|b_k' + \mu_{k,k-1}b_{k-1}'\|_2^2 \ge \frac{3}{4} \|b_{k-1}'\|_2^2 \hfill (\widetilde{*}) \]
First note that we already have $|\mu_{k,k-1}| \le 1/2$ (from $(*)$). Let $l$ be the largest integer such that $|\mu_{k,l}| > 1/2$. (Recall $\mu_{i,j}$ is only defined for $1 \le j < i \le n$). Let $r \in \mathbb{Z}$ be such that $|\mu_{k,l} - r| \le 1/2$. We then replace $b_k$ with $b_k - rb_l$. Thus $\mu_{k,l}$ becomes $\mu_{k,l} - r$.

\noindent This change affects other $\mu_{k,j}$ for $j < l$, but it does not affect $b_i, b_i', \mu_{i,j}$ for $i < k$. It also doesn't affect $(\widetilde{*})$, as $l < k-1$. We repeat this process until we obtain that the basis satisfies $|\mu_{k,j}| \le 1/2$ for all $1 \le j \le k-1$. The final basis satisfies Property$(k+1)$.
To prove that the algorithm terminates, we need the following lemma, the proof of which will be on the next exercise sheet.
\begin{lemma}{Lower Bound for $\Delta$}{2.21}
    Let $\{ b_1, \dots, b_n \}$ be a basis of $\Lambda$. Then
    $$ \Delta(b_1, \dots, b_n) \ge \frac{m(\Lambda)^{n^2}}{4^{n^2} V_n^{2n}} $$
    where $\Delta(b_1, \dots, b_n) = D_1\cdots D_n$ with $D_i = \det \left( [b_1 \dots b_i]^T [b_1 \dots b_i] \right) = \det \left( [b_1' \dots b_i']^T [b_1' \dots b_i'] \right) \forall_{1 \leq i \leq n}$, $m(\Lambda) = \min \{ \langle x, x \rangle : x \in \Lambda \setminus \{0\} \}$ and $V_n$ is the volume of the unit ball in $\mathbb{R}^n$
\end{lemma}

\end{pf}
\noindent We can now prove that the algorithm terminates.
\begin{pf}[Proof that the LLL-Algorithm terminates]
It is clear that $D_n = (\det \Lambda)^2$. Furthermore,
\[ D_i = \|b_1'\|_2^2 \cdots \|b_i'\|_2^2 = (\det \Lambda_i)^2 \]
where $\Lambda_i = [b_1 \dots b_i v_{i+1} \dots v_n] \mathbb{Z}^n$ with $\langle b_j, v_t \rangle = 0$ and $\{ v_{i+1}, \dots, v_n \}$ orthonormal (Gram-Schmidt sheet 3).

It can be checked that in both cases 0 and 2, $D_1, \dots, D_n$ (and hence $\Delta$) do not change. (It only involves modification of the form $b_k \to b_k - cb_i$ for some $i < k$).

In case 1, only $D_{k-1}$ changes:
\begin{align*} D_{k-1} = \|b_1'\|_2^2 \cdots \|b_{k-1}'\|_2^2 \to \quad & \|c_1'\|_2^2 \cdots \|c_{k-1}'\|_2^2 \\
= \quad & \|b_1'\|_2^2 \cdots \|c_{k-1}'\|_2^2 \\
< \quad & \frac{3}{4} \|b_1'\|_2^2 \cdots \|b_{k-1}'\|_2^2 \\
= \quad &\frac{3}{4} D_{k-1, \text{old}} \end{align*}
But all the other $D_i$ do not change, which implies that $\Delta_{\text{new}} < \frac{3}{4} \Delta_{\text{old}}$.
But $\Delta$ cannot get arbitrarily small due to Lemma \ref{lem:2.21}.
\end{pf}

\paragraph{Conclusion:} The RHS depends only on $\Lambda$ and $n$. Therefore, in the algorithm, Case 1 can only occur finitely many times (i.e. $k \to k-1$ only bounded number of times), and hence $k$ must reach $n+1$ and the algorithm terminates!

\begin{example}
  Let $\alpha_1, \dots, \alpha_n \in \R$, not all in $\Q$. Also let $\epsilon > 0$ be small. Suppose we want to find $q \in \N$ and $x_1, \dots, x_n \in \Z$ such that \begin{equation}\label{eq:ex_lll_1}
    |q\alpha_i-x_i| \leq \epsilon \forall_{1 \leq i \leq n}. \tag{1}
  \end{equation}
  Recall that in the proof of Corollary \ref{cor:2.7}, we proved that given $Q > 1$, $\exists_{(x_1, \dots, x_n) \in \Zprim{n+1}}: 0 < q \leq Q \land |q\alpha_i-x_i| \leq Q^{-\nicefrac{1}{n}}$. Applying this with $Q^{-\nicefrac{1}{n}}=\epsilon$, we can solve (\ref{eq:ex_lll_1}) with \begin{equation}\label{eq:ex_lll_2} 0 < q \leq \epsilon^{-n}.\tag{2}\end{equation}
  Next, let's see what we get using LLL. Consider
  \begin{equation*}
    \Lambda = \begin{pmatrix}
      & & & -\alpha_1 \\
      & \1_n & & \vdots \\
      & & & -\alpha_n \\
      0 & \cdots & 0 & \delta
    \end{pmatrix}\Z^{n+1},
  \end{equation*}
  with $\delta > 0$ to be chosen later. Then $\Lambda = \{ (x_1-q\alpha_1, \dots, x_n-q\alpha_n, \delta q): x_1, \dots, x_n,q \in \Z\}$. Recall that LLL efficiently produces a reduced basis $\{ b_1, \dots, b_{n+1}\}$ of $\Lambda$, satisfying
  \[ \|b_1\|_2 \leq 2^{\frac{n}{4}}\det\Lambda^{\frac{1}{n+1}} = 2^{\frac{n}{4}}\delta^{\frac{1}{n+1}}.\]
  This was Proposition \ref{prop:2.19}(iii), one of the properties of a reduced basis. Let us choose $\delta$ such that
  \[ 2^{\frac{n}{4}}\delta^{\frac{1}{n+1}} = \epsilon \iff \delta = \epsilon^{n+1}2^{\frac{-n(n+1)}{4}}.\]
  Let us denote
  \[ b_1 = \begin{pmatrix}
    x_1-q\alpha_1 \\
    \vdots \\
    x_n-q\alpha_n \\
    \delta q
  \end{pmatrix}\]
  (assume $q \geq 0$ without loss of generality). Then
  \[ \sqrt{(x_1-q\alpha_1)^2+\dots+(x_n-q\alpha_n)^2+(\delta q)^2} \leq \epsilon,\]
  which implies
  \[ |x_i-q\alpha_i| \leq \epsilon \forall_{1 \leq i \leq n} \text{ and } 0 \leq q < \frac{\epsilon}{\delta} = \epsilon^{-n}2^{\frac{n(n+1)}{4}}.\]
  Note that if $q=0$, we get a contradiction.
\end{example}
\chapter{The Bombieri-Pila Determinant Method}
We have done a lot with lattice points inside a central symmetric convex body. In this section, we will count lattice points on a curve\footnote{Professor Yamagishi: For our intents and purposes, this is just a zero set of a two-variable polynomial. No schemes, sheaves, connected. None of these Algebraic Geometry adjectives. Any Algebraic Geometers here? No one? Yeah, no one's taking that course.}. Our goal in this chapter is to prove the following.
\begin{theorem}{Bombieri-Pila, 1989}{3.1}
  Let $F \in \R[x,y]$ be an irreducible polynomial of degree $d \geq 2$. If $N$ is sufficiently large with respect only to $d$, then
  \[ \#\{ (x,y) \in [1,N]^2 \cap \Z^2: F(x,y) = 0\} \leq (\log N)^{\O(d)}N^{\frac{1}{d}}.\]
\end{theorem}
\noindent \textbf{Remarks:} The implicit constant in $\O(d)$ is absolute. In particular, it does not depend on the coefficients of $F$. Therefore, the bound holds uniformly over all irreducible $F \in \R[x,y]$ of degree $d$. Also, it is best possible up to the $(\log N)^{\O(d)}$ term (consider $F(x,y) = y-x^{\alpha}$).
\begin{pf}[Very rough proof idea]
Draw the rectangle as well as the solution curve. Cover the solution curve with a small number of tiles that look like
\[ \{ (x,f(x)): x \in [a,b]\}\]
for some $a < b \in \R$ and $f$ as well as its derivatives decaying rapidly. As we shall see, we will be able to bound the integer points of
\[ \#\Z^2 \cap \{ (x,f(x)): x \in [a,b]\}\]
efficiently.
\end{pf}
\section{Technical Preparation}
Some notation for this chapter: The set $\mathcal{M} \subset \R[x,y]$ is a finite set of monomials, id est
\[ \mathcal{M} = \{ x^{j_1}y^{j_2}: \vec j = (j_1, j_2) \in S\}\]
with a finite $S \subset \Nzero^2$. We also denote $S(\mathcal{M}) = S$. Further, we define
\[ J_1 = J_1(\mathcal{M}) = \sum_{x^{j_1}y^{j_2} \in \mathcal{M}} j_1 \text{ and } J_2 = J_2(\mathcal{M}) = \sum_{x^{j_1}y^{j_2} \in \mathcal{M}} j_2.\]
\begin{lemma}{Technical Preparation for Bombieri-Pila}{3.2}
  Let $\mathcal{M}$ be as above and $D = \#\mathcal{M} \geq 2$. Assume that $I \subseteq [0,N]$ is a closed interval and $f \in \mathcal{C}^{D-1}(I)$. Further suppose that $X > 0$ and $\delta \geq \frac{1}{N}$ satisfy that
  \[ \left| \frac{f^{(i)}(x)}{i!}\right| \leq X\delta^i \forall_{0 \leq i < D, x \in I}.\]
  If
  \[ |I| < \frac{1}{4}\delta^{-1}\left((2N)^{J_1}(DX)^{J_2}\right)^{-\frac{1}{\binom{D}{2}}},\]

But this lemma allows us to find $P \in \R[x,y]$ such that $\Z^2 \cap \{ (x,f(x)): x \in I\} \subseteq V(P)$ with good quantitative properties ($\deg P$ and $|I|$).

  \begin{lemma}{Rank Criterion}{3.3}
    Let $\mathcal{M}$ be as above and $\vec z_1, \dots, \vec z_t \in \R^2$. If $\rank((\vec z_i^{\vec j})_{\stackrel{1 \leq i \leq t}{\vec j \in S(\mathcal{M})}}) < \# \mathcal{M}$, then there exists
    \[ 0 \neq P \in \Span_{\R}(\mathcal{M}): \vec z_1, \dots, \vec z_t \in V(P).\]
  \end{lemma}
  \begin{pf}
    Suppose that the rank of the matrix is $r$. Without loss of generality, we let
    \[ \rank((\vec z_i^{\vec j})_{\stackrel{1 \leq i \leq r}{\vec j \in S(\mathcal{N})}}) = r, \text{ where } \mathcal{N} \subseteq \mathcal{M} \text{ with } \#\mathcal{N} = r.\]
    By assumption, we have $r = \#\mathcal{N} < \#\mathcal{M}$, which means that there exists $\vec k = (k_1, k_2) \in S(\mathcal{M}) \setminus S(\mathcal{N})$. Consider $P \in \R[x,y]$ defined by
    \begin{align*}
      P(x,y) = \quad & \det A(x,y) \\
      \coloneq \quad & \det\begin{pmatrix}
        (x^{j_1}y^{j_2})_{\vec j \in S(\mathcal{N})} & x^{k_1}y^{k_2} \\
        & \vec z_1^{\vec k} \\
        (\vec z_i^{\vec j})_{\stackrel{1 \leq i \leq r}{\vec j \in S(\mathcal{N})}} & \vdots \\
        & \vec z_r^{\vec k}
      \end{pmatrix} \\
      = \quad & \sum_{\vec j \in S(\mathcal{N}) \cup \{ \vec k\}} \epsilon_{\vec j}\det(A_{\vec j})x^{j_1}y^{j_2},
    \end{align*}
    where $\epsilon_{\vec j} \in \{ \pm 1\}$ and $A_{\vec j}$ is the matrix obtained from $A(x,y)$ by removing the top row and the column $\vec j$. We know that at least one of the $\det A_{\vec j}$ is nonzero, which ensures that our polynomial is nonzero. Therefore, we have
    \[ P \in \Span_{\R}(\mathcal{N} \cup \{ x^{k_1}y^{k_2}\}) \subseteq \Span_{\R}(\mathcal{M}).\]
    Finally, if $1 \leq i \leq r$, then $A(\vec z_i)$ has a repeated row, which means that $P(\vec z_i) = 0$. If $r+1 \leq i \leq t$, then $\det A(\vec z_i) = 0$ because if $\rank A(\vec z_i) = r+1$, this contradicts that
    \[ r = \rank((\vec z_i^{\vec j})_{\stackrel{1 \leq i \leq t}{\vec j \in S(\mathcal{M})}}).\]
    Hence, we have $\vec z_1, \dots, \vec z_t \in V(P).$
\end{pf}

\begin{lemma}{Generalised Vandermonde Bound}{3.4}
  Let $n \geq 1$, let $I \subseteq \R$ be a closed interval, and suppose $f_1, \dots, f_n \in \mathcal{C}^{n-1}(I)$. Let $A_{i,j} \geq 0$ be such that
  \[ \max_{x \in I}\left| \frac{f_j^{(i-1)}(x)}{(i-1)!}\right| \leq A_{i,j} \forall_{1 \leq i,j \leq n}.\]
  Then, for any $x_1, \dots, x_n \in I$, we have
  \[ \left|\det(f_j(x_i))_{1 \leq i,j \leq n}\right| \leq \prod_{1 \leq m < i \leq n}|x_i-x_m| \cdot \sum_{\sigma \in S_n}\prod_{i=1}^n A_{i,\sigma(i)},\]
  where $S_n$ denotes the group of permutations of $\{ 1, \dots, n\}$.
\end{lemma}
\begin{pf}
  If $x_1, \dots, x_n$ are not distinct, then the result is trivial. Therefore, assume they are distinct.

  First suppose $1 \leq i,j \leq n$. We can find $\xi_{i,j} \in I$ such that
  \begin{equation}\label{eq:3.4_divided_difference}
    \frac{f_j^{(i-1)}(\xi_{i,j})}{(i-1)!}
    = \sum_{l=1}^i f_j(x_l)\prod_{\substack{1 \leq m \leq i \\ m \neq l}}\frac{1}{x_l-x_m}.
    \tag{1}
  \end{equation}
  To see this, consider
  \[ g(\xi) = \sum_{l=1}^i f_j(x_l)\prod_{\substack{1 \leq m \leq i \\ m \neq l}}\frac{\xi-x_m}{x_l-x_m}.\]
  We have $\deg g<i$ and $g(x_\ell)=f_j(x_\ell)$ for all $1 \leq \ell \leq i$. Hence, by repeated applications of Rolle's theorem, there exists $\xi_{i,j} \in I$ such that
  \[ \frac{d^{i-1}}{d\xi^{i-1}}(g-f_j)(\xi_{i,j}) = 0.\]
  Thus,
  \[ f_j^{(i-1)}(\xi_{i,j}) = g^{(i-1)}(\xi_{i,j}) = \sum_{l=1}^i f_j(x_l)(i-1)!\prod_{\substack{1 \leq m \leq i \\ m \neq l}}\frac{1}{x_l-x_m},\]
  as desired.

  For $1 \leq l \leq i$, let
  \[ G_{l,i}(x_1,\dots,x_i) = (i-1)!\prod_{\substack{1 \leq m \leq i \\ m \neq l}}\frac{1}{x_l-x_m}.\]
  Then from (1) we get
  \[ f_j^{(i-1)}(\xi_{i,j}) = \sum_{l=1}^i f_j(x_l)G_{l,i}(x_1,\dots,x_i)\]
  for some $\xi_{i,j}\in I$. Hence,
  \begin{align*}
    &\det\left(f_j^{(i-1)}(\xi_{i,j})\right)_{1 \leq i,j \leq n} \\
    =\quad & \det\left(\sum_{l=1}^i f_j(x_l)G_{l,i}(x_1,\dots,x_i)\right)_{1 \leq i,j \leq n} \\
    =\quad & \det\begin{pmatrix}
      G_{1,1} & 0 & 0 & \cdots \\
      G_{1,2} & G_{2,2} & 0 & \cdots \\
      G_{1,3} & G_{2,3} & G_{3,3} & \cdots \\
      \vdots & \vdots & \vdots & \ddots
    \end{pmatrix}
    \cdot \det(f_j(x_i))_{1 \leq i,j \leq n} \\
    =\quad & \prod_{i=1}^n G_{i,i}(x_1,\dots,x_i)\cdot \det(f_j(x_i))_{1 \leq i,j \leq n}.
  \end{align*}
  Therefore,
  \begin{align*}
    \left|\det(f_j(x_i))_{1 \leq i,j \leq n}\right|
    &= \prod_{i=1}^n |G_{i,i}(x_1,\dots,x_i)|^{-1}\cdot \left|\det\left(f_j^{(i-1)}(\xi_{i,j})\right)_{1 \leq i,j \leq n}\right| \\
    &\leq \prod_{i=1}^n \frac{\prod_{1 \leq m < i}|x_i-x_m|}{(i-1)!}\cdot \sum_{\sigma \in S_n}\prod_{i=1}^n |f_{\sigma(i)}^{(i-1)}(\xi_{i,\sigma(i)})| \\
    &\leq \prod_{1 \leq m < i \leq n}|x_i-x_m|\cdot \sum_{\sigma \in S_n}\prod_{i=1}^n A_{i,\sigma(i)}. \qedhere
  \end{align*}
\end{pf}

\begin{lemma}{}{3.7}
  Suppose that $F \in \R[x,y]$ is irreducible of degree $d$ over $\R$, with $\partial_yF(x,y) \not\equiv 0$. Let $I \subseteq \R$ be a closed interval (not a point), $k \geq 1$ and $f \in \mathcal{C}^{\infty}(I)$ satisfy $F(x,f(x)) = 0 \forall_{x \in I}$.
  Suppose that $f$ is \emp{NOT} a degree $k$ polynomial. Then, we have
  \[ \#\{ x \in I: f^{(k)}(x) = c\} \leq 2kd^2 \forall_{c \in \R^\times}.\]
\end{lemma}
\begin{pf}
  We basically take the $k$-th derivative of $F(x,f(x)) = 0$ and see what happens. We show by induction that for each $k \geq 1$,
  \[ \exists_{H_k \in \R[x,y]: \deg(H_k) \leq D_k = (k-1)(2d-3)+d-1}: H_k(x,f(x)) + \partial_yF(x,f(x))^{2k-1}f^{(k)}(x) = 0 \forall x \in I. \hfill (*).\]
  For $k=1$, we have
  \[ \partial_xF(x,f(x)) + \partial_yF(x,f(x))f^{(1)}(x) \equiv 0 \forall_{x \in I}.\]
  Therefore, $(*)$ holds with $H_k(x,y) = \partial_xF(x,y)$ and $\deg(H_k) \leq d-1 = D_1$. Assume that $(*)$ hold for some $k \geq 1$. Take the derivative of $(*)$ to get
  \begin{align*} 0 \equiv \quad & \partial_xH_k(x,f(x)) + \partial_yH_k(x,f(x))f'(x) + \partial_yF(x,f(x))^{2k-1}f^{(k+1)} + \\
  \quad & (2k-1)F_y(x,f(x))^{2k-2}f^{(k)}(\partial^2_{xy}F(x,f(x))+\partial^2_yF(x,f(x))f'(x)).\end{align*}
  Multiply by $\partial_yF(x,f(x))^2$ (and simplify the notation) to get
  \[ 0 = \partial_yF^2(\partial_xH_k + \partial_yH_kf') + \partial_yF^{2k+1}f^{(k+1)} + (2k-1)\partial_yF^{2k}f^{(k)}(\partial_{xy}^2F + \partial_y^2f').\]
  Now use $(*)$ ($H_k = -\partial_yF^{2k-1}f^{(k)}$ and $\partial_xF = -\partial_yFf'$) to deduce that
  \begin{align*} 0 = \quad & \partial_yF^2\partial_xH_k -\partial_xF\partial_yF\partial_yH_k + F_y^{2k+1}f^{(k+1)}-(2k-1)H_k\underbrace{\partial_yF(\partial_{xy}^2F + \partial_y^2Ff')}_{= \partial_yF\partial_{xy}^2F-\partial_xF\partial^2_{y}}F \\
  = \quad & H_{k+1}(x,f(x)) + \partial_yF(x,f(x))^{2k+1}f^{k+1}(x) \forall x \in I.\end{align*}
  Check that $\deg H_{k+1} \leq D_k + 2d-3 = D_{k+1}$. This proves $(*)$ for $k+1$.
  It follows from $(*)$ that, given $x \in I$ and $c \in \R^{\times}$,
  \[ f^{(k)}(x) = c \implies R_c(x,f(x)) = 0,\]
  where $R_c(x,y) = H_k(x,y) + \partial_yF(x,y)^{2k-1}c$. Check that $\deg R_c \leq 2kd$. Therefore,
  \begin{align*}
    & \{ x \in I: f^{(k)}(x) = c\} \\
    \subseteq \quad & \{ x \in I: R_c(x,f(x)) = 0\} \\
    \subseteq \quad & \{ x \in I: R_c(x,y) = 0, F(x,y) = 0 \text{ for some } y \in \R\}.
  \end{align*}
  This implies that (under the assumption that $F \nmid R_c$, which we will prove later)
  \begin{align*}
    & \#\{ x \in I: f^{(k)}(c)\} \leq \#\{ (x,y) \in \R^2: F(x,y) = 0, R_c(x,y) = 0\} \\
    \stackrel{F \nmid R_c, \text{ Thm. \ref{thm:3.6}} }{\leq} \quad & \deg(F)\deg(R_c) \\
    \leq \quad & d2kd \\
    = \quad & 2kd^2,
  \end{align*}
  as desired. It remains to show that $F \nmid R_c$. Let us suppose otherwise. We know that $F(x,f(x)) = 0$ implies $R_c(x,f(x)) = 0$, where we have
  \[ R_c(x,f(x)) = H_k(x,f(x)) + \partial_yF(x,f(x))^{2k-1}c \stackrel{(*)}{=} -\partial_yF(x,f(x))^{2k-1}(f^{(k)}(x)-c).\]
  By the same argument as before,
  \begin{align*}
    & \#\{ x \in I: \partial_yF(x,f(x)) = 0\} \\
    \leq \quad & \{ (x,y) \in \R^2: F(x,y) = \partial_yF(x,y) = 0\} \\
    \stackrel{F \nmid \partial_yFf}{\leq} \quad & \deg(F)\deg(\partial_yF) \\
    \leq \quad & d(d-1).
  \end{align*}
  Hence, $f^{(k)}(x) = c \neq 0 \forall_{x \in I}$, except for at mosst $d^2$ points. Thus, $f$ is of degree $k$. \Lightning \\
  This completes the proof.
\end{pf}

\begin{lemma}{}{3.9}
  Let $\ell \geq d \geq 2$ and $I \subseteq [0,N]$ be a closed intervall. Fix $f \in \mathcal{C}^{\infty}(I)$ with $F(x,f(x)) = 0 \forall_{x \in I}$, $F$ being of degree $d$ and irreducible over $\R$. Suppose that $N > 0$ and $\delta \geq \frac{1}{\lambda(I)}$ satisfy
  \[ \max_{x \in I}\left| \frac{f^{(i)}(x)}{i!}\right| \leq N\delta^i \forall_{0 \leq i \leq k = d(\ell-d+1)}.\]
  Then
  \[ \#\{ (x,f(x)): x \in I\} \cap \Z^2 \ll (d\ell)^cN^{\frac{1}{d} + \O\left(\frac{1}{\ell}\right)}\delta\lambda(I)\]
  for some absolute $c, \ell, d > 0$.
\end{lemma}
\begin{pf}
  Let $i_0$ be the largest $i$ such that $x^{d-i}y^i$ is a monomial appearing in $F$. Define $\mathcal M = \{ x^{j_1}y^{j_2}: d \leq j_1+j_2 \leq \ell, x^{d-i_0}y^{i_0} \nmid x^{j_1}y^{j_2}\}$. We shall now compute $\#\mathcal M$. We have
  \begin{align*}
    \#\mathcal M = \quad & \sum_{d \leq h \leq \ell} \#\{ (j_1,j_2) \in S(\mathcal M): h = j_1+j_2\}.
  \end{align*}
  Fix $d \leq h \leq \ell$. Clearly, $j_2 = h-j_1$. Now $x^{d-i_0}y^{i_0} \nmid x^{j_1}y^{j_2}$ implies that $d-i-0 > j_1$ or $i_0 > j_2$. Hence, $0 \leq j_1$ or $h-i_0<j_1 \leq h$. Since $h \geq d \geq i_0$, the number of choices for $j_1$ is $d-i_0+i_0 = d$. Therefore, \[\#\mathcal{M} =  \sum_{d \leq h \leq \ell} d = d(\ell-d+1) = k = d\ell + \O(d^2).\]
  Also, \begin{align*} J_1(M) + J_2(M) = \quad & \sum_{d \leq h \leq \ell} dh \\
  = \quad & d\left(\frac{\ell(\ell+1)}{2}-\frac{d(d-1)}{2}\right) \\
  \leq \quad & \frac{\ell^2d}{2} + \O(d\ell) \\
  = \quad & \frac{k^2}{2} + \O\left(\frac{k^2}{\ell}\right).\end{align*}
  This implies that
  \begin{equation}\label{eq:lem3.9:1}
    \frac{J_1(M)+J_2(M)}{\binom{k}{2}} \leq \frac{1}{d} + \O\left(\frac{1}{\ell}\right). \tag{1}
  \end{equation}
  Next, we show that $F \nmid P$ for any $0 \not\equiv P \in \Span_{\R}(M)$. Suppose that $F \mid P$. This implies that $P = FG$ with $\deg G = d'$. Let $i_0'$ be the largest $i$ such that $x^{d-i}y^i$ is a monomial appearing in $G$. Then $x^{d+d'-i_0-i_0'}y^{i_0+i_0'}$ is a monomial appearing in $P$. Therefore, we have
  \[ x^{d+d'-i_0-i_0'} \in \mathcal M,\]
  which is a contradiction because $x^{d-i_0}y^{i_0}$ divides this monomial. \Lightning \\
  Under the hypothesis of this lemma, it follows from Lemma \ref{lem:3.2*} that
  \begin{equation}\label{eq:lem3.9:2} \{ (x,f(x)): x \in I\} \cap \Z^2 \subseteq \bigcup_{P \in \mathfrak{P}} V(P),\tag{$*$}\end{equation}
  where the union is over $\P \subseteq \Span_\R(\mathcal M)$ of size \begin{align*}\leq \quad &  4\delta\lambda(I)((2N)^{J_1(M)}(kN)^{J_2(M)})^{\frac{1}{\binom{k}{2}}} + 1 \\
  \stackrel{\text{(\ref{eq:lem3.9:1})}}{\leq} \quad &  4\delta\lambda(I)N^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)}k^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)}+1.\end{align*}
  We therefore have
  \begin{align*}
    \#\{ (x,f(x)): x \in I\} \cap \Z^2 \leq \quad & \sum_{P \in \P} \#\{ (x,y) \in \Z^2: F(x,y)=P(x,y)=0\} \\
    \stackrel{\text{Bézout, } F \nmid P}{\leq} \quad & \sum_{P \in \P} \deg(F)\deg(P) \\
    \ll \quad & (4\delta\lambda(I)N^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)}+k^{\frac{1}{d}+\O\left(\frac{1}{\ell}+1\right)})d\ell \\
    \ll \quad & (d\ell)^c\delta\lambda(I)N^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)},
  \end{align*}
  where we used that the left hand side of (\ref{eq:lem3.9:2}) is contained in $V(F) \cap \Z^2$. Also recall that $k = d\ell + \O(d^2)$.
\end{pf}
\begin{lemma}{}{3.10}
  Let $k \geq 1$, $0 < \delta <  1$, and $X > 0$. Further assume that $I$ is a closed bounded interval and $f \in \mathcal{C}^k(I)$. If
  \[ \max_{x \in I}\left|\frac{f^{(i)}(x)}{i!}\right| \leq X\delta^i \forall_{0 \leq i < k}\]
  and
  \[ \min_{x \in I}\left|\frac{f^{(k)}(x)}{k!}\right| \geq X\delta^k,\]
  we have $\lambda(I) \leq \frac{2}{\delta}$.
\end{lemma}
\begin{pf}
  Let $I = [a,b]$. Then there exists $\xi \in I$ such that
  \[ f(b)-f(a) = \sum_{i=1}^{k-1}\frac{f^{(i)}(a)}{i!}(b-a)^i + \frac{f^{(k)}(\xi)}{k!}(b-a)^k.\]
  Therefore, we have
  \[ X\delta^k\lambda(I)^k \leq \left| \frac{f^{(k)}(\xi)}{k!}(b-a)^k\right| \leq 2X+\sum_{i=1}^{k-1}X\delta^i\lambda(I)^i.\]
  This implies that
  \[ \delta^k\lambda(I)^k \leq 2 + \sum_{i=1}^{k-1}\delta^i\lambda(I)^i.\]
  Write $\lambda = \delta\lambda(I)$. Then
  \[ 2 + \sum_{i=1}^{k-1}\lambda^i = 2 + \frac{\lambda^k-\lambda}{\lambda-1}.\]
  If $\lambda >2$, then $\lambda^k \leq 2 + \lambda^k-\lambda$, which implies $\lambda \leq 2$ \Lightning. Therefore, we have $\lambda \leq 2$, i.e. $\lambda(I) \leq \frac{2}{\delta}$.
\end{pf}

\section{Number of lattice points on a curve}
\begin{theorem}{Number of lattice points on a curve}{3.11}
  Let $d \geq 2$ and let $N \geq 1$ be sufficiently large in terms of $d$. Let $I \subseteq [0,N]$ be a closed interval and let $f \in \mathcal{C}^{\infty}(I)$ satisfy
  \[ |f'(x)| \leq 1 \forall_{x \in I}.\]
  Suppose that there exists an irreducible polynomial $F \in \R[x,y]$ of degree $d$ such that
  \[ F(x,f(x)) = 0 \forall_{x \in I}.\]
  Then
  \[ \#\left(\{(x,f(x)): x \in I\}\cap \Z^2\right) \leq (\log N)^{\O(d)}N^{\frac{1}{d}}.\]
\end{theorem}
\begin{proof}
  For $T > 1$, define $g(T)$ to be the maximum of $2$ and the supremum of
  \[ \#\left(\{(x,h(x)): x \in J\}\cap\Z^2\right)\]
  over all closed intervals $J \subseteq [0,T]$ with $\lambda(J) \geq 1$, all $h \in \mathcal{C}^{\infty}(J)$ with $|h'(x)| \leq 1$ for all $x \in J$, and all irreducible $G \in \R[x,y]$ of degree $d$ such that $G(x,h(x))=0$ on $J$. For $0 < T \leq 1$, set $g(T)=2$.

  We first prove the following recursive estimate.
  \begin{claim}{}{}
    Given any integer $\ell \geq d$, there exists $K = K(d,\ell)$ with
    \[ 2 \leq K \leq \ell^{\O(1)}\]
    such that, for all $T > 0$,
    \[ g(T) \leq K^{\O(d)}T^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)} + K g(K^{-2d}T),\]
    where the implicit constants are absolute.
  \end{claim}

  We show that the claim implies the theorem. After increasing the implicit constants, the claim gives absolute constants $c_0,c_1,c_2>0$ such that
  \[ g(T) \leq K^{c_1d}T^{\frac{1}{d}+\frac{c_2}{\ell}} + Kg(K^{-2d}T) \quad \text{and} \quad K \leq \ell^{c_0}.\]
  Since $N$ is sufficiently large in terms of $d$, we may choose $\ell = \lceil \log N\rceil \geq d$. Iterating the estimate, we obtain
  \[ g(N) \leq K^{c_1d}N^{\frac{1}{d}+\frac{c_2}{\ell}}\sum_{j=0}^{n-1}K^{-j} + K^ng(K^{-2nd}N),\]
  where $n$ is chosen such that $K^{-2nd}N \leq 1 < K^{-2d(n-1)}N$. Thus $g(K^{-2nd}N) \leq 2$ and $K^n \leq KN^{\frac{1}{2d}}$. Hence
  \[ g(N) \ll K^{c_1d}N^{\frac{1}{d}+\frac{c_2}{\ell}} + KN^{\frac{1}{2d}} \ll \ell^{\O(d)}N^{\frac{1}{d}+\frac{c_2}{\ell}} \ll (\log N)^{\O(d)}N^{\frac{1}{d}}.\]
  If $\lambda(I)<1$, then the desired count is at most $2$, so the theorem follows from the bound for $g(N)$.

  It remains to prove the claim.

  \smallskip
  \noindent\emph{Proof of the claim.}
    Let $J \subseteq [0,T]$ be a closed interval with $\lambda(J)=X$, where $1 \leq X \leq T$, and let $h \in \mathcal{C}^{\infty}(J)$ be such that $|h'(x)| \leq 1$ for all $x \in J$ and $G(x,h(x))=0$ for some irreducible $G \in \R[x,y]$ of degree $d$.

    Since the range of $h$ on $J$ has length at most $X$, after replacing $h$ by $h+y_0$ for a suitable $y_0 \in \Z$, and $G(x,y)$ by $G(x,y-y_0)$, we may assume
    \[ |h(x)| \leq X \forall_{x \in J}.\]
    This replacement does not change the number of lattice points on the graph.

    Fix $\ell \geq d$, put $k=d(\ell-d+1)$, and choose a parameter $\delta \in \left(\frac{2}{X},1\right)$ for the moment. Apply Lemma \ref{lem:3.8} with $A_i=i!X\delta^i$ for $1 \leq i \leq k$. This partitions $J$ into $\O(k^2d^2)$ intervals $J_v$ such that, for each $0 \leq i \leq k$ and each $J_v$, either
    \[ \left|\frac{h^{(i)}(x)}{i!}\right| \leq X\delta^i \forall_{x \in J_v}\]
    or
    \[ \left|\frac{h^{(i)}(x)}{i!}\right| \geq X\delta^i \forall_{x \in J_v}.\]
    Here the assertion for $i=0$ is always in the first case.

    Pick one of the intervals $J_v$. If $\lambda(J_v)\geq \frac{1}{\delta}$ and the first alternative holds for all $0 \leq i \leq k$, then Lemma \ref{lem:3.9}, after translating $J_v$ horizontally into $[0,X]$, gives
    \[ \#\left(\{(x,h(x)):x\in J_v\}\cap\Z^2\right) \ll (d\ell)^cX^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)}\delta\lambda(J_v).\]

    Otherwise, either $\lambda(J_v)<\frac{1}{\delta}$, or there exists $1 \leq k_0 \leq k$ such that
    \[ \left|\frac{h^{(i)}(x)}{i!}\right| \leq X\delta^i \forall_{x \in J_v}, \ 0 \leq i < k_0,\]
    and
    \[ \left|\frac{h^{(k_0)}(x)}{k_0!}\right| \geq X\delta^{k_0}\forall_{x \in J_v}.\]
    In the latter case, Lemma \ref{lem:3.10} gives $\lambda(J_v)\leq \frac{2}{\delta}$. Hence every interval in this second case contributes at most $g\left(\frac{2}{\delta}\right)$ lattice points.

    Combining the two cases, we get
    \begin{align*}
      \#\left(\{(x,h(x)):x\in J\}\cap\Z^2\right)
      \ll \quad & (d\ell)^cX^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)}\delta X \\
      &+ C_1d^2(d\ell)^2g\left(\frac{2}{\delta}\right)
    \end{align*}
    for some absolute constant $C_1>0$.

    Choose
    \[ K=C_1d^2(d\ell)^2.\]
    Increasing $C_1$ if necessary, we have $2 \leq K \leq \ell^{\O(1)}$. If $2K^{2d}<X$, set
    \[ \delta=\frac{2K^{2d}}{X}.\]
    Then
    \[ \#\left(\{(x,h(x)):x\in J\}\cap\Z^2\right)
      \leq K^{\O(d)}X^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)} + Kg(K^{-2d}X).\]
    Since $X \leq T$ and $g$ is increasing, this is
    \[ \leq K^{\O(d)}T^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)} + Kg(K^{-2d}T).\]
    On the other hand, if $2K^{2d}\geq X$, then
    \[ \#\left(\{(x,h(x)):x\in J\}\cap\Z^2\right) \leq X+1 \leq 2K^{2d}+1,\]
    and therefore
    \[ \#\left(\{(x,h(x)):x\in J\}\cap\Z^2\right)
      \leq K^{\O(d)}T^{\frac{1}{d}+\O\left(\frac{1}{\ell}\right)} + Kg(K^{-2d}T),\]
    since $T \geq 1$. Taking the supremum over all admissible $J,h,G$ proves the claim, and hence the theorem.
\end{proof}

\begin{theorem}{Construction of an Auxiliary Polynomial}{4.6}
  Let $0 < \epsilon < 1$. Let $\alpha$ be an algebraic integer of degree $n$ with the minimal polynomial $f(t) = t^n + a_{n-1}t^{n-1} + \dots + a_0 \in \Z[t]$. Let $m > 8n^2\epsilon^{-2}$ be an integer, and let $r_1, \dots, r_m \in \N$. Then there exists $P(x_1, \dots, x_m) \in \Z[x_1, \dots, x_m]$ with $J = J(r_1, \dots, r_m)$ such that
  \begin{enumerate}[(i)]
    \item $P \not\equiv 0$,
    \item $\operatorname{ind}_{\vec r}(P, (\alpha, \dots, \alpha)) \geq \tfrac{1}{2}m(1-\epsilon)$ and
    \item $\|P\| \leq (4(H(\alpha)+1))^{r_1 + \dots + r_m}$.
  \end{enumerate}
\end{theorem}
\noindent In other words, we have constructed a polynomial $P(x_1, \dots, x_m)$ with a zero of large order at $(\alpha, \dots, \alpha)$!
\begin{pf}
  Let us denote
  \[ P(x_1, \dots, x_m) = \sum_{\vec j \in J(r_1, \dots, r_m)}c(j_1, \dots, j_m)x_1^{j_1}\cdots x_m^{j_m},\]
  with $c(j_1, \dots, j_m)$ integers to be determined. Set
  \[ N = (r_1 + 1)\cdots(r_m+1) = \#J(r_1, \dots, r_m).\]
  We want $P^{[\vec k]}(\alpha, \dots, \alpha) = 0$ for all $\vec k \in (\Z_{\geq 0})^m$ satisfying
  \[ \sum_{i=1}^m\frac{k_i}{r_i} < \tfrac{1}{2}(m-\epsilon m),\]
  since this implies $\operatorname{ind}_{\vec r}(P, (\alpha, \dots, \alpha)) \geq \tfrac{1}{2}(m-\epsilon m)$. From Lemma \ref{lem:4.4}, we know
  \[ \alpha^{\ell} = a_{n-1}^{(\ell)}\alpha^{n-1} + \dots + a_0^{(\ell)}, \tag{$*$}\]
  with integers $|a_j^{(\ell)}| \leq (H(\alpha)+1)^{\ell}$. Then
  \[ P^{[\vec k]}(\alpha, \dots, \alpha) = \sum_{\substack{k_i \leq j_i \leq r_i \\ (1 \leq i \leq m)}}\binom{j_1}{k_1}\cdots\binom{j_m}{k_m}c(j_1, \dots, j_m)\alpha^{(j_1-k_1) + \dots + (j_m-k_m)}.\]
  By substituting ($*$) and collecting the powers of $\alpha$, we obtain
  \[ = \sum_{h=0}^{n-1}\alpha^h\underbrace{\sum_{\vec j}\binom{j_1}{k_1}\cdots\binom{j_m}{k_m}c(j_1, \dots, j_m)a_h^{(j_1-k_1+\dots+j_m-k_m)}}_{=\, Q_{\vec k, h} \,\in\, \Z}.\]
  Since $\{ 1, \alpha, \dots, \alpha^{n-1}\}$ is linearly independent over $\Q$, we have
  \[ P^{[\vec k]}(\alpha, \dots, \alpha) = 0 \iff 0 = Q_{\vec k, 0} = \dots = Q_{\vec k, n-1}.\]
  Also, $|a_h^{(\ell)}| \leq (H(\alpha)+1)^{\ell}$, $\binom{j_i}{k_i} \leq 2^{j_i} \leq 2^{r_i}$ and $j_1 - k_1 + \dots + j_m - k_m \leq r_1 + \dots + r_m$. Now we view $0 = Q_{\vec k, 0} = \dots = Q_{\vec k, n-1}$ as a system of linear equations in the $c(\vec j)$, with integer coefficients whose absolute values are
  \[ \leq (H(\alpha)+1)^{r_1 + \dots + r_m}\cdot 2^{r_1 + \dots + r_m}.\]
  It follows from Lemma \ref{lem:4.5} that
  \begin{align*}
    \#\left\{ \vec k \in J(r_1, \dots, r_m): \sum_{i=1}^m\frac{k_i}{r_i} \leq \tfrac{1}{2}(m-m\epsilon)\right\} &\leq \frac{\sqrt{2m}}{m\epsilon}(r_1+1)\cdots(r_m+1) \\
    &= \frac{\sqrt 2}{\sqrt m \cdot \epsilon}N < \frac{N}{2n},
  \end{align*}
  since $m > 8n^2\epsilon^{-2}$. Then
  \[ 0 = Q_{\vec k, 0} = \dots = Q_{\vec k, n-1} \quad \left(\forall \vec k \in J(r_1, \dots, r_m) \text{ s.t. } \sum_{i=1}^m\frac{k_i}{r_i} \leq \tfrac{1}{2}(m-m\epsilon)\right)\]
  is a system of $M$ linear equations over $\Z$ in $N$ variables, where (for each $\vec k$ there are $n$ equations)
  \[ M < \frac{N}{2n}\cdot n = \frac{N}{2} < N = \#J(r_1, \dots, r_m),\]
  and $|\text{coeff.}| \leq H = (2\cdot(H(\alpha)+1))^{r_1 + \dots + r_m}$. We now invoke the following result, which we will prove next class.
  \begin{quote}
    \textbf{(Siegel's Lemma)} Given a system of $M$ linear equations over $\Z$ in $N$ variables with $N > M$, whose coefficients satisfy $|\text{coeffs}| \leq H$, there exists a solution $\vec x \in \Z^N \setminus \{ 0\}$ such that
    \[ \max_{1 \leq i \leq N}|x_i| \leq (N\cdot H)^{\frac{M}{N-M}}.\]
  \end{quote}
  Therefore, we can choose $c(\vec j) \in \Z$ for all $\vec j \in J(r_1, \dots, r_m)$, not all of them zero, such that
  \[ \underbrace{0 = Q_{\vec k, 0} = \dots = Q_{\vec k, n-1}}_{\iff\, P^{[\vec k]}(\alpha, \dots, \alpha) = 0} \quad \left(\forall \vec k \in J(r_1, \dots, r_m) \text{ s.t. } \sum_{i=1}^m\frac{k_i}{r_i} \leq \tfrac{1}{2}(m-m\epsilon)\right)\]
  and
  \begin{align*}
    0 < \max|c(\vec j)| &\leq (NH)^{\frac{M}{N-M}} \\
    &\leq N\cdot H \tag{$2M < N$} \\
    &\leq (4\cdot(H(\alpha)+1))^{r_1 + \dots + r_m}. \qedhere
  \end{align*}
\end{pf}

\begin{theorem}{Siegel's lemma}{4.7}
  Let $N>M$ and $A = (a_{ij}) \in M_{M \times N}(\Z)$ with $\max_{i,j}|a_{ij}| \leq H$. Then there exists $x \in \Z^N$ such that $Ax = 0$ with $0 < \max_{1 \leq i \leq N}|x_i| \leq (NH)^{\frac{M}{N-M}}$.
\end{theorem}
\begin{pf}
  For each $1 \leq i \leq M$, define $L_i(x) = \sum_{j=1}^N a_{ij}x_j$. Denote
  \[ -C_i = \sum_{j=1}^N \min\{a_{ij},0\}\]
  and
  \[ D_i = \sum_{j=1}^N \max\{ a_{ij}, 0\}.\]
  Clearly, $0 \leq C_i+D_i \leq NH$. Given any $y \in \Z^N \cap [0,B]^N$, we have $-C_iB \leq L_i(y) \leq D_iB \forall_{1 \leq i \leq M}$. Let $B \in \N$. Then
  \begin{align*} & \#\{ (L_1(y), \dots, L_M(y)): y \in \Z^N \cap [0,B]^N\} \\
  \leq \quad & \prod_{i=1}^M D_iB+C_iB+1 \\
  \leq \quad & \prod_{i=1}^M (NHB+1) \\
  = \quad & (NHB+1)^M.\end{align*}
  If $\#\{ y \in \Z^N \cap [0,B]^N\} = (B+1)^N > (NHB+1)^M$, then there exist distinct $y_1, y_2 \in \Z^N \cap [0,B]^N$ such that $L_i(y_1) = L_i(y_2) \forall_{1 \leq i \leq M}$. Hence, $L_i(y_1-y_2) = 0 \forall_{1 \leq i \leq M}$ and $Ax = 0$ with $0 \neq x= y_1-y_2$ and $x \in \Z^N \cap [-B,B]^N$. Finally, check that \[ B = \lf (NH)^{\frac{M}{N-M}}\rf\]
  does the job.
\end{pf}
Let's recall some notation.
\[ P(x_1, \dots, x_m) = \sum_{j \in J} C(j)x_1^{j_1}x_m^{j_m},\]
\[ J = J(r_1, \dots, r_m) = \{ 0, \dots, r_1\} \times \cdots \times \{ 0, \dots, r_m\},\]
\[ \|P\| = \max_{j \in J}|c(j)|\]
and

Also define
\[ \text{ind}_{s}(P,\alpha) = \min\left\{ \sum_{i=1}^m \frac{k_i}{s_i}: P^{[k]}(\alpha) \neq 0, k \in J\right\}\]
for $s \in \N^m$. Last time we proved Theorem \ref{thm:4.6}. That is, we constructed a polynomial over $\Z$ with a zero of \glqq large\grqq \ order at $(\alpha, \dots, \alpha)$. \begin{theorem}{}{4.8}
  Let $0 < \delta < \frac{1}{20}$ and $0 < \epsilon < \frac{\delta}{100}$ and $\alpha$ an algebraic integer as in Theorem \ref{thm:4.6}. Suppose that $\frac{a_h}{q_h} \in \Q$ with $1 \leq h \leq m$ fulfil
  \[ \left| \alpha - \frac{a_h}{q_h}\right| < \frac{1}{q_h^{2+\delta}}\]
  and $q_h^{\epsilon} > 64(H(\alpha)+1)\max\{ 1, |\alpha|\} \forall_{1 \leq h \leq m}$. Also suppose thta  $m > 8n^2\epsilon^{-2}$ and $r_1, \dots, r_m \in \N$ are such that $r_1\log q_1 \leq r_h\log q_h \leq (1+\epsilon)r_1\log q_1 \forall_{1 \leq h \leq m}$. Then the integer polynomial $P(x_1, \dots, x_m)$ constructed in Theorem \ref{thm:4.6} satisfies $\text{ind}_r\left(P,\left(\frac{a_1}{q_1}, \dots, \frac{a_m}{q_m}\right)\right) \geq \frac{\delta_m}{8}$.
\end{theorem}
\begin{pf}
  We begin by applying Theorem \ref{thm:4.6} to get $P(x_1, \dots, x_m) \in \Z[x_1, \dots, x_m]$ with $\text{ind}_r(P, (\alpha, \dots, \alpha)) \geq \frac{1}{2}m(1-\epsilon)$. We want to show: Let $k_1, \dots, k_m \in \Nzero$ fulfil
  \[ \sum_{h=1}^m \frac{k_i}{r_i} < \frac{\delta_m}{8}. \hfill (*)\]
  Then $P^{[k]}\left(\frac{a_1}{q_1}, \dots, \frac{a_m}{q_m}\right) = 0$ $\left(\text{because this implies } \text{ind}_r\left(p, \left(\frac{a_1}{q_1}, \dots, \frac{a_m}{q_m}\right)\right) \geq \frac{\delta_m}{8}\right)$. Let $k$ satisfy $(*)$ and denote $T(x_1, \dots, x_n) = P^{[k]}(x_1, \dots, x_m)$. It follows from Lemma \ref{lem:4.2} and Theorem \ref{thm:4.6} that $T(x_1, \dots, x_m) \in \Z[x_1, \dots, x_m]$, $\|T\| = (8H(\alpha)+8)^{r_1+r_m}$ and $\deg_{x_h}T \leq r_h \forall_{1 \leq h \leq m}$. Given any $i = (i_1, \dots, i_m) \in J(r_1, \dots, r_m)$, we have
  \begin{align*} \left| T^{[i]}(\alpha, \dots, \alpha)\right| \leq \quad & 2^{r_1+\dots+r_m}\|T^{[i]}|\max\{1, |\alpha|\}^{r_1+\dots+r_m} \\
  \leq \quad & (32H(\alpha)+32)^{r_1+\dots+r_m}\max\{ 1, |\alpha|\}^{r_1+\dots+r_m}.\end{align*}
  because $\|T^{[i]}\| \leq 2^{r_1+\dots+r_m}\|T\|$. Next, by Lemma \ref{lem:4.3}, we have
  \begin{align*} \text{ind}_r(T, (\alpha, \dots, \alpha)) \geq \quad & \text{ind}_r(P,(\alpha, \dots, \alpha)) - \sum_{n=1}^m\frac{k_h}{r_h} \\
  \stackrel{\text{Thm. \ref{thm:4.6}}}{>} \quad & \frac{1}{2}m(1-\epsilon)-\frac{\delta m}{8} \\
  = \quad & \frac{1}{2}m\left(1-\epsilon-\frac{\delta}{4}\right) \\
  > \quad & \frac{1}{2}m\left(1-\frac{\delta}{3}\right).\end{align*}
  Therefore, $T^{[i]}(\alpha, \dots, \alpha) = 0$ if
  \[ \sum_{h=1}^m \frac{i_h}{r_h} \leq \frac{1}{2}m\left(1-\frac{\delta}{3}\right).\]
  Now, let us put $\beta_h = \alpha - \frac{a_h}{q_h} \forall_{1 \leq h \leq m}$. Then $|\beta_h| < \frac{1}{q_h^{2+\delta}}$. By the Taylor series expansion, we get
  \begin{align*}
    T\left(\frac{a_1}{q_1}, \dots, \frac{a_m}{q_m}\right) = \quad & \sum_{i \in J(r_1, \dots, r_m)} T^{[i]}(\alpha, \dots, \alpha)\beta_1^{i_1}\cdots\beta_m^{i_m} \\
    = \quad & \sum_{\stackrel{i \in J}{\sum_{h=1}^m \frac{i_h}{r_h} > \frac{1}{2}m(1-\frac{\delta}{3})}} T^{[i]}(\alpha, \dots, \alpha)\beta_1^{i_1}\cdots\beta_h^{i_h}. \hfill (**)
  \end{align*}
  For any $i$ in the sum,
  \begin{align*}
    -\log|\beta_1^{i_1}\cdots\beta_m^{i_m}| > \quad & (2+\delta)\sum_{h=1}^m\frac{i_h}{r_h}(r_h\log q_h) \\
    \geq \quad & (2+\delta)r_1\log(q_1)\underbrace{\sum_{h=1}^m \frac{i_h}{r_h}}_{> \frac{1}{2}m\left(1-\frac{\delta}{3}\right)} \\
    > \quad & \left(1+\frac{\delta}{2}\right)\left(1-\frac{\delta}{3}\right)mr_1\log(q_1) \\
    \geq \quad & \left(1+\frac{\delta}{2}\right)\left(1-\frac{\delta}{3}\right)\sum_{h=1}^m r_h\log(q_h)\frac{1}{1+\epsilon}\\
    > \quad & (1+\epsilon)^2\sum_h r_h\log(q_h)\frac{1}{1+\epsilon}
  \end{align*}
  for $0 < \delta < \frac{1}{20}$ and $0 < \epsilon < \frac{\delta}{100}$, which implies $|\beta_1\cdots\beta_m| <(q_1^{r_1}\cdots q_m^{r_m})^{-1-\epsilon}$. Hence, $(**)$ becomes
  \begin{align*} & \Nzero \ni \left|q_1^{r_1}\cdots q_m^{r_m}T\left(\frac{a_1}{q_1}, \dots, \frac{a_m}{q_m}\right)\right| \\
  < \quad & \sum_{i \in J}|T^{[i]}(\alpha, \dots, \alpha)|(q_1^{r_1}\cdots q_m^{r_m})^{-\epsilon} \\
  \leq \quad & 2^{r_1+\dots+r_m}\left[(32H(\alpha)+32)\max\{ |\alpha|, 1\}\right]^{r_1+\dots+r_m}(q_1^{r_1}\cdots q_m^{r_m})^{-\epsilon} \\
  \leq \quad & 1.\end{align*}
  This proves the assumption.
\end{pf}

\begin{theorem}{}{4.10}
  Let $m \in \N$, $0 < \epsilon < \frac{1}{12}$, and set $\omega = \omega(m,\epsilon) = 24\cdot2^{-m}\left(\frac{\epsilon}{12}\right)^{\exp(\ln(2)(m-1))} < 1$. Let $r_1, \dots, r_m \in \N$ be such that $\omega r_h \geq r_{h+1} \forall_{1 \leq h \leq m-1}$. Let $q_1, \dots, q_m \in \N$ be such that $q_h^{r_h} \geq q_1^{r_1}$ and $q_h^{\omega} \geq 2^{3m} \forall_{1 \leq h \leq m}$. Suppose $P \in \Z[x_1, \dots, x_m] \setminus \{ 0\}$ is such that $\deg_{x_h} P \leq r_h \forall_{1 \leq h \leq m}$ and $\| P\| \leq q_1^{\omega r_1}$. Then
  \[ \ind_r\left(P, \left(\frac{a_1}{q_1}, \dots, \frac{a_m}{q_m}\right)\right) \leq \epsilon,\]
  provided $\gcd(a_h,q_h)=1 \forall_{1 \leq h \leq m}$.
\end{theorem}
\begin{pf}[Rough idea of the proof]
  Say $P(x_1, x_2)$ has large order of vanishing at $\left(\frac{a_1}{q_1},\frac{a_2}{q_2}\right)$. Write $P(x_1, x_2) = \phi_0(x_2)+\phi_1(x_2)x_1 + \dots + \phi_r(x_2)x_1^r$. Define
  \[ 0 \not\equiv W(x_1,x_2) = \underbrace{\det(\Delta_i\phi_j(x_2))}_{U(x_2) \in \Z[x_2]}\underbrace{\det(\Delta_ix_1^{j-1})}_{V(x_1) \in \Z[x_1]}.\]
  A large order of vanishing of $P$ at $\left(\frac{a_1}{q_1},\frac{a_2}{q_2}\right)$ would imply the same for $W$, which in turn implies the same for $V$ at $\frac{a_1}{q_1}$ or $U$ at $\frac{a_2}{q_2}$. Hence, $(q_1x_1-a_1)^N \mid V(x_1)$. But if $\|V\| < q_1^N$, then $V \equiv 0$. Thus, we have $W \equiv 0$, which contradicts our assumptions.
\end{pf}
\noindent We shall now give a real proof.
\begin{pf}
  We prove this by induction on $m$. For $m=1$, suppose that
  \[ 0 = P\left(\frac{a_1}{q_1}\right) = \dots = P^{(t-1)}\left(\frac{a_1}{q_1}\right) \text{ and } P^{(t)}\left(\frac{a_1}{q_1}\right) \neq 0.\]
  Here $\gcd(a_1,q_1) = 1$ and $q_1 > 0$. Write $P(x) = \left(x-\frac{a_1}{q_1}\right)^tT(x)$ for some $T \in \Q[x]$. Hence,
  \[ P(x) = (q_1x-a_1)^t\underbrace{(q_1^{-t}T(x))}_{\in \Z[x]}\]
  by Gauß's lemma. Thus, $q_1^t \leq \|P\| \leq q_1^{\omega r_1}$, which implies $t \leq \omega r_1$. Since $m=1$, $\omega(1,\epsilon) = \epsilon$. Since $m=1$, $\omega(1,\epsilon) = \epsilon$, we have
  \[ \ind_{r_1}\left(P, \frac{a_1}{q_1}\right) = \frac{t}{r_1} \leq \omega = \epsilon,\]
  which completes the base case. Now suppose that the whole thing holds for all all $m' < m \in \N$. Let us write
  \[ P(x_1, \dots, x_m) = \sum_{1 \leq j \leq k}\phi_j(x_1, \dots, x_{m-1})\psi_j(x_m),\]
  where $\phi_j, \psi_j$ are $\Q$-polynomials. We fix such a representation with $k$ minimal. We know that $k \leq r_m+1$ by choosing $\psi_j(x_m) = x_m^{j-1}$. Let us prove that $\phi_j$ and $\psi_j$ have two properties.
  \begin{claim}{}{}
    The set $\{ \phi_1, \dots, \phi_k\}$ is linearly independent over $\Q$.
  \end{claim}
  \begin{pf}[Proof of the claim]
    Suppose that there exist $c_1, \dots, c_k \in \Q$, not all zero, such that
    \[ c_1\phi_1 + \dots + c_k\phi_k \equiv 0.\]
    Assume that $c_k = -1$ without loss of generality. This implies $\phi_k = c_1\phi_1 + \dots + c_{k-1}\phi_{k-1}$. Now substitute this into $P$ and get
    \[ P(x_1, \dots, x_m) = \sum_{j=1}^{k-1}\phi_j(x_1, \dots, x_{m-1})(\psi_j(x_m)+c_j\psi_k(x_m)),\]
    which contradicts the minimality of $k$ \Lightning.
  \end{pf}
  \begin{claim}{}{}
    The set $\{ \psi_1, \dots, \psi_k\}$ is linearly independent over $\Q$.
  \end{claim}
  \begin{pf}[Proof of this claim]
    The same argument.
  \end{pf}
  Since $\psi_1, \dots, \psi_k$ are polynomials over $\Q$ and linearly independent over $\Q$, it follows from Proposition \ref{prop:4.9} that
  \[ U_1(x_m) = \left(\diff[i]{}{x_m}\psi_j(x_m)\right)_{1 \leq i,j \leq k} \not\equiv 0.\]
  Similarly, there exist $\widetilde{\Delta}_1, \dots, \widetilde{\Delta}_k$ such that
  \[ \widetilde{\Delta}_i = \frac{\partial^{i_1+\dots+i_{m-1}}}{\partial x_1^{i_1}\cdots\partial_{m-1}^{i_{m-1}}}\]
  with $\ord\widetilde{\Delta} = i_1+\dots+i_{m-1} \leq i-1 \leq k-1 \leq r_m$. Then $V_1(x_1, \dots, x_{m-1}) = \det(\widetilde{\Delta}_i\phi_j)_{1 \leq i,j \leq k} \not\equiv 0$. We also denote
  \[ \iota(i) = (i_1, \dots, i_{m-1}).\]
  Next, we define
  \[ W_1(x_1, \dots, x_m) = \det\left(\widetilde{\Delta}_i\frac{\partial^{j-1}}{\partial x_m^{j-1}}P(x_1, \dots, x_m)\right)_{1 \leq i,j \leq k}.\]
  Since
  \begin{align*} & \widetilde{\Delta}_i\frac{\partial^{j-1}}{\partial x_m^{j-1}}P(x_1, \dots, x_m) \\
  = \quad & \widetilde{\Delta}_i\frac{\partial^{j-1}}{\partial x_m^{j-1}}\sum_{h=1}^k\phi_h(x_1, \dots, x_{m-1})\phi_h(x_m) \\
  = \quad & \sum_{h=1}^k \widetilde{\Delta}_i\phi_h\diff[j-1]{}{x_m}\psi_h(x_m),\end{align*}
  we have
  \begin{align*}
    W_1(x_1, \dots, x_m) = \quad & \det\left((\widetilde{\Delta}_i\phi_h)\left(\diff[j-1]{}{x_m}\psi_h\right)^t\right) \\
    = \quad & V_1(x_1, \dots, x_{m-1})U_1(x_m) \\
    \not\equiv \quad & 0.
  \end{align*}
  Then let
  \begin{align*}
    W(x_1, \dots, x_m) = \quad & \frac{W_1(x_1, \dots, x_m)}{\prod_{i=1}^k \iota(i)!\prod_{j=1}^k (j-1)!} \\
    = \quad & \det(P^{[\iota(i), j-1]})_{1 \leq i,j \leq k}.
  \end{align*}
  This si still a polynomial over $\Z$ because $P^{[j]} \in \Z[x_1, \dots, x_m]$ for any $j$. We can deduce from Gauß's lemma that
  \[ W(x_1, \dots, x_m) = V(x_1, \dots, x_{m-1})U(x_m)\]
  with $U,V$ polynomials over $\Z$ (in fact $c_1U_1 = U$ and $c_2V_1 = V$ with $c_1,c_2 \in \Q$). Since $\deg_{x_h}P^{[j]} \leq r_h \forall_{1 \leq h \leq m}, j \in \Nzero^m$, $\deg_{x_h}w \leq r_hk \forall_{1 \leq h \leq m}$. Similarly, $\deg_{x_h}V \leq r_hk \forall_{1 \leq h \leq m-1}$ and $\deg_{x_m}U \leq r_mk$. Also,
  \[ \| P^{[j]}\| \leq 2^{r_1+\dots+r_m}\|P\| \leq 2^{r_1+\dots+r_m}q_1^{r\omega_1}.\]
  It follows that
  \[ \|W\| \leq k!\left(2^{r_1+\dots+r_m}\max_{i,j}\|P^{[(\iota(i),j-1)]}\|\right)^k\]
  by the bound for the number of summand in $P^{[j]}$. This is in turn less than or equal to
  \begin{align*}
    & k!\left(4^{r_1+\dots+r_m}q_1^{r\omega}\right)^k \\
    \leq \quad & 2^{kr_m}\left(4^{r_1+\dots+r_m}q_1^{r\omega}\right)^k \\
    \leq \quad & \left(8^{r_1+\dots+r_m}q_1^{r_1\omega}\right)^k \\
    \leq \quad & 8^{mr_1k}q_1^{r_1\omega k} \\
    \leq \quad & (q_1^\omega)^{r_1k}q_1^{r_1\omega k} \\
    = \quad & q_1^{2\omega r_1k}.
  \end{align*}
  Recall that
  \begin{align*}
    W = \quad & U(x_m)V(x_1, \dots, x_{m-1}) \\
    = \quad & \sum_{\ell} u_{\ell}x_m^{\ell}\sum_{k}v_kx_1^{k_1}\cdots x_{m-1}^{k_{m-1}} \\
    = \quad & \sum_{(\ell, k)} u_{\ell}v_kx_1^{k_1}\cdots x_{m-1}^{k_{m-1}}x_m^{\ell}.
  \end{align*}
  This implies that $\|U\|, \|V\| \leq \|W\| \leq q_1^{2\omega r_1k}$.
  Index of $V$: We consider $\ind_{kr}\left(V, \left(\frac{a_1}{q_1}, \dots, \frac{a_{m-1}}{q_{m-1}}\right)\right)$. By the inductive hypothesis, we have
  \begin{align*}
    \ind_{kr}\left(V, \left(\frac{a_1}{q_1}, \dots, \frac{a_{m-1}}{q_{m-1}}\right)\right) \leq \epsilon_0,
  \end{align*}
  provided that $\|V\| \leq q_1^{2\omega r_1k} \leq q_1^{\omega(m-1,\epsilon_0)r_1k}$. Since $\omega = \omega(m,\epsilon) = \frac{1}{2}\omega\left(m-1, \frac{\epsilon^2}{12}\right)$, we choose $\epsilon_0 = \frac{\epsilon^2}{12}$. Note that all of the other hypotheses are satisfied. Thus,
  \begin{align*}
    & \ind_{kr}\left(V, \left(\frac{a_1}{q_1}, \dots, \frac{a_{m-1}}{q_{m-1}}\right)\right) \leq \frac{\epsilon^2}{12} \\
    \implies & \ind_r\left(V, \left(\frac{a_1}{q_1}, \dots, \frac{a_{m-1}}{q_{m-1}}\right)\right) \leq \frac{k\epsilon^2}{12},
  \end{align*}
\end{pf}
